38  void ak_gf64_mul_uint64( ak_pointer z, ak_pointer x, ak_pointer y )
39 {
40  int i = 0, n = 0;
41  ak_uint64 zv = 0, t = ((ak_uint64 *)y)[0], s = ((ak_uint64 *)x)[0];
42 
43  t = ((ak_uint64 *)y)[0];
44  for( i = 0; i < 64; i++ ) {
45 
46  if( t&0x1 ) zv ^= s;
47  t >>= 1;
48  n = (s >> 63);
49  s <<= 1;
50  if( n ) s ^= 0x1B;
51  }
52  ((ak_uint64 *)z)[0] = zv;
53 }

61  void ak_gf128_mul_uint64( ak_pointer z, ak_pointer x, ak_pointer y )
62 {
63  int i = 0, n = 0;
64  ak_uint64 t, s0 = ((ak_uint64 *)x)[0], s1 = ((ak_uint64 *)x)[1];
65 
66  /* обнуляем результирующее значение */
67  ((ak_uint64 *)z)[0] = 0; ((ak_uint64 *)z)[1] = 0;
68 
69  /* вычисляем произведение для младшей половины */
70  t = ((ak_uint64 *)y)[0];
71  for( i = 0; i < 64; i++ ) {
72 
73  if( t&0x1 ) { ((ak_uint64 *)z)[0] ^= s0; ((ak_uint64 *)z)[1] ^= s1; }
74  t >>= 1;
75  n = s1 >> 63;
76  s1 <<= 1; s1 ^= ( s0 >> 63 ); s0 <<= 1;
77  if( n ) s0 ^= 0x87;
78  }
79 
80  /* вычисляем произведение для старшей половины */
81  t = ((ak_uint64 *)y)[1];
82  for( i = 0; i < 63; i++ ) {
83 
84  if( t&0x1 ) { ((ak_uint64 *)z)[0] ^= s0; ((ak_uint64 *)z)[1] ^= s1; }
85  t >>= 1;
86  n = s1 >> 63;
87  s1 <<= 1; s1 ^= ( s0 >> 63 ); s0 <<= 1;
88  if( n ) s0 ^= 0x87;
89  }
90 
91  if( t&0x1 ) {
92  ((ak_uint64 *)z)[0] ^= s0;
93  ((ak_uint64 *)z)[1] ^= s1;
94  }
95 }

107  void ak_gf64_mul_pcmulqdq( ak_pointer z, ak_pointer x, ak_pointer y )
108 {
109  #ifdef LIBAKRYPT_HAVE_BUILTIN_SET_EPI64X
110  const __m128i gm = _mm_set_epi64x( 0, 0x1B );
111  __m128i xm = _mm_set_epi64x( 0, ((ak_uint64 *)x)[0] );
112  __m128i ym = _mm_set_epi64x( 0, ((ak_uint64 *)y)[0] );
113 
114  __m128i cm = _mm_clmulepi64_si128( xm, ym, 0x00 );
115  __m128i cx = _mm_set_epi64x( 0, cm[1] );
116 
117  xm = _mm_clmulepi64_si128( cx, gm, 0x00 ); xm[1] ^= cx[0];
118  ym = _mm_set_epi64x( 0, xm[1] );
119  xm = _mm_clmulepi64_si128( ym, gm, 0x00 );
120 
121  ((ak_uint64 *)z)[0] = cm[0]^xm[0];
122 
123  #else
124  __m128i gm, xm, ym, cm, cx;
125 
126  gm.m128i_u64[0] = 0x1B; gm.m128i_u64[1] = 0;
127  xm.m128i_u64[0] = ((ak_uint64 *)x)[0]; xm.m128i_u64[1] = 0;
128  ym.m128i_u64[0] = ((ak_uint64 *)y)[0]; ym.m128i_u64[1] = 0;
129 
130  cm = _mm_clmulepi64_si128( xm, ym, 0x00 );
131  cx.m128i_u64[0] = cm.m128i_u64[1]; cx.m128i_u64[1] = 0;
132 
133  xm = _mm_clmulepi64_si128( cx, gm, 0x00 );
134  xm.m128i_u64[1] ^= cx.m128i_u64[0];
135  ym.m128i_u64[0] = xm.m128i_u64[1]; ym.m128i_u64[1] = 0;
136  xm = _mm_clmulepi64_si128( ym, gm, 0x00 );
137 
138  ((ak_uint64 *)z)[0] = cm.m128i_u64[0]^xm.m128i_u64[0];
139  #endif
140 }

148  void ak_gf128_mul_pcmulqdq( ak_pointer z, ak_pointer a, ak_pointer b )
149 {
150 #ifdef LIBAKRYPT_HAVE_BUILTIN_SET_EPI64X
151  __m128i am = _mm_set_epi64x( ((ak_uint64 *)a)[1], ((ak_uint64 *)a)[0] );
152  __m128i bm = _mm_set_epi64x( ((ak_uint64 *)b)[1], ((ak_uint64 *)b)[0] );
153 
154  /* умножение */
155  __m128i cm = _mm_clmulepi64_si128( am, bm, 0x00 ); // c = a0*b0
156  __m128i dm = _mm_clmulepi64_si128( am, bm, 0x11 ); // d = a1*b1
157  __m128i em = _mm_clmulepi64_si128( am, bm, 0x10 ); // e = a0*b1
158  __m128i fm = _mm_clmulepi64_si128( am, bm, 0x01 ); // f = a1*b0
159 
160  /* приведение */
161  ak_uint64 x3 = dm[1];
162  ak_uint64 D = dm[0] ^ em[1] ^ fm[1] ^ (x3 >> 63) ^ (x3 >> 62) ^ (x3 >> 57);
163 
164  cm[0] ^= D ^ (D << 1) ^ (D << 2) ^ (D << 7);
165  cm[1] ^= em[0] ^ fm[0] ^ x3 ^ (x3 << 1) ^ (D >> 63) ^ (x3 << 2) ^ (D >> 62) ^ (x3 << 7) ^ (D >> 57);
166 
167  ((ak_uint64 *)z)[0] = cm[0];
168  ((ak_uint64 *)z)[1] = cm[1];
169 
170 #else
171  __m128i am, bm, cm, dm, em, fm;
172  ak_uint64 x3, D;
173 
174  am.m128i_u64[0] = ((ak_uint64 *)a)[0]; am.m128i_u64[1] = ((ak_uint64 *)a)[1];
175  bm.m128i_u64[0] = ((ak_uint64 *)b)[0]; bm.m128i_u64[1] = ((ak_uint64 *)b)[1];
176 
177  /* умножение */
178  cm = _mm_clmulepi64_si128( am, bm, 0x00 ); // c = a0*b0
179  dm = _mm_clmulepi64_si128( am, bm, 0x11 ); // d = a1*b1
180  em = _mm_clmulepi64_si128( am, bm, 0x10 ); // e = a0*b1
181  fm = _mm_clmulepi64_si128( am, bm, 0x01 ); // f = a1*b0
182 
183  /* приведение */
184  x3 = dm.m128i_u64[1];
185  D = dm.m128i_u64[0] ^ em.m128i_u64[1] ^ fm.m128i_u64[1] ^ (x3 >> 63) ^ (x3 >> 62) ^ (x3 >> 57);
186 
187  cm.m128i_u64[0] ^= D ^ (D << 1) ^ (D << 2) ^ (D << 7);
188  cm.m128i_u64[1] ^= em.m128i_u64[0] ^ fm.m128i_u64[0] ^ x3 ^ (x3 << 1) ^ (D >> 63) ^ (x3 << 2) ^ (D >> 62) ^ (x3 << 7) ^ (D >> 57);
189 
190  ((ak_uint64 *)z)[0] = cm.m128i_u64[0];
191  ((ak_uint64 *)z)[1] = cm.m128i_u64[1];
192 
193  #endif
194 }

219  "using null pointer to authentication key");

221  __func__, "using key with very large block size" );
222  /* инициализация значением и ресурс */
223  if(( authenticationKey->key.flags&ak_skey_flag_set_key ) == 0 )

227  return ak_error_message( ak_error_low_key_resource, __func__, "using key with low key resource");
228

230  "using null pointer to initial vector");

232  __func__, "using initial vector with unexpected length");

234  ctx->aflag = 0;

236  memset( &ctx->sum, 0, 16 ); /* очищаем по максимуму */
237  memset( &ctx->mulres, 0, 16 );
238  memset( &ctx->h, 0, 16 );
239  memset( &ctx->zcount, 0, 16 );
240  memset( ivector, 0, 16 );
241  memcpy( ivector, iv, iv_size ); /* копируем нужное количество байт */

247  authenticationKey->encrypt( &authenticationKey->key, ivector, &ctx->zcount );
248  authenticationKey->key.resource.counter--;
249 
250  return ak_error_ok;

279  int ak_mgm_context_authentication_update( ak_mgm_ctx ctx,
280  ak_bckey authenticationKey, const ak_pointer adata, const size_t adata_size )
281 {
282  ak_uint8 temp[16], *aptr = (ak_uint8 *)adata;
283  size_t absize = authenticationKey->ivector.size;
284  ak_int64 resource = 0,
285  tail = (ak_int64) adata_size%absize,
286  blocks = (ak_int64) adata_size/absize;
287 
288  /* проверка возможности обновления */
289  if( ctx->aflag&0x1 )
290  return ak_error_message( ak_error_wrong_block_cipher_function, __func__ ,
291  "using this function with previously closed aead context");
292  /* ни чего не задано => ни чего не обрабатываем */
293  if(( adata == NULL ) || ( adata_size == 0 )) return ak_error_ok;
294 
295  /* проверка ресурса ключа */
296  if( authenticationKey->key.resource.counter <= (resource = blocks + (tail > 0)))
297  return ak_error_message( ak_error_low_key_resource, __func__, "using key with low key resource");
298  else authenticationKey->key.resource.counter -= resource;
299 
300  /* теперь основной цикл */
301  if( absize == 16 ) { /* обработка 128-битным шифром */
302 
303  ctx->abitlen += ( blocks << 7 );
304  for( ; blocks > 0; blocks--, aptr += 16 ) { astep128( aptr ); }
305  if( tail ) {
306  memset( temp, 0, 16 );
307  memcpy( temp+absize-tail, aptr, (size_t)tail );
308  astep128( temp );
309  /* закрываем добавление ассоциированных данных */
310  ctx->aflag |= 0x1;
311  ctx->abitlen += ( tail << 3 );
312  }
313  } else { /* обработка 64-битным шифром */
314 
315  ctx->abitlen += ( blocks << 6 );
316  for( ; blocks > 0; blocks--, aptr += 8 ) { astep64( aptr ); }
317  if( tail ) {
318  memset( temp, 0, 8 );
319  memcpy( temp+absize-tail, aptr, (size_t)tail );
320  astep64( temp );
321  /* закрываем добавление ассоциированных данных */
322  ctx->aflag |= 0x1;
323  ctx->abitlen += ( tail << 3 );
324  }
325  }
326 
327  return ak_error_ok;
328 }

343  ak_buffer ak_mgm_context_authentication_finalize( ak_mgm_ctx ctx,
344  ak_bckey authenticationKey, ak_pointer out, const size_t out_size )
345 {
346  ak_uint128 temp;
347  ak_pointer pout = NULL;
348  ak_buffer result = NULL;
349  size_t absize = authenticationKey->ivector.size;
350 
351  /* проверка запрашиваемой длины iv */
352  if(( out_size == 0 ) || ( out_size > absize )) {
353  ak_error_message( ak_error_wrong_length, __func__, "unexpected length of integrity code" );
354  return NULL;
355  }
356  /* проверка длины блока */
357  if( absize > 16 ) {
358  ak_error_message( ak_error_wrong_length, __func__, "using key with large block size" );
359  return NULL;
360  }
361 
362  /* традиционная проверка ресурса */
363  if( authenticationKey->key.resource.counter <= 0 ) {
364  ak_error_message( ak_error_low_key_resource, __func__, "using key with low key resource");
365  return NULL;
366  } else authenticationKey->key.resource.counter--;
367 
368  /* закрываем добавление шифруемых данных */
369  ctx->pflag |= 0x1;
370 
371  /* формируем последний вектор из длин */
372  if( absize&0x10 ) {
373  temp.q[0] = ctx->pbitlen;
374  temp.q[1] = ctx->abitlen;
375  astep128( temp.b );
376 
377  } else { /* теперь тоже самое, но для 64-битного шифра */
378 
379  if(( ctx->abitlen > 0xFFFFFFFF ) || ( ctx->pbitlen > 0xFFFFFFFF )) {
380  ak_error_message( ak_error_overflow, __func__, "using an algorithm with very long data" );
381  return NULL;
382  }
383  temp.w[0] = (ak_uint32) ctx->pbitlen;
384  temp.w[1] = (ak_uint32) ctx->abitlen;
385  astep64( temp.b );
386  }
387 
388  /* определяем указатель на область памяти, в которую будет помещен результат вычислений */
389  if( out != NULL ) pout = out;
390  else {
391  if(( result =
392  ak_buffer_new_size( authenticationKey->ivector.size )) != NULL ) pout = result->data;
393  else ak_error_message( ak_error_get_value( ), __func__ , "wrong creation of result buffer" );
394  }
395 
396  /* последнее шифрование и завершение работы */
397  if( pout != NULL ) {
398  authenticationKey->encrypt( &authenticationKey->key, &ctx->sum, &ctx->sum );
399  memcpy( pout, ctx->sum.b+absize-out_size, out_size );
400  } else ak_error_message( ak_error_out_of_memory, __func__ ,
401  "incorrect memory allocation for result buffer" );
402  return result;
403 }
