53  int ak_skey_create( ak_skey skey, size_t size, size_t isize )
54 {
55  int error = ak_error_ok;
56  if( skey == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
57  "using a null pointer to secret key" );
58  if( size == 0 ) return ak_error_message( ak_error_zero_length, __func__,
59  "using a zero length for key size" );
60  if( isize == 0 ) return ak_error_message( ak_error_zero_length, __func__,
61  "using a zero length for integrity code" );
62  /* Инициализируем данные базовыми значениями */
63  if(( error = ak_buffer_create_function_size( &skey->key,
64  /* в настоящий момент для выделения памяти используются стандартные функции */
65  malloc, free,
66  size )) != ak_error_ok ) {
67  ak_error_message( error, __func__ ,"wrong creation a secret key buffer" );
68  ak_skey_destroy( skey );
69  return error;
70  }
71  if(( error = ak_buffer_create_size( &skey->mask, size )) != ak_error_ok ) {
72  ak_error_message( error, __func__ ,"wrong creation a key mask buffer" );
73  ak_skey_destroy( skey );
74  return error;
75  }
76  if(( error = ak_buffer_create_size( &skey->icode, isize )) != ak_error_ok ) {
77  ak_error_message( error, __func__ ,"wrong creation a integrity code buffer" );
78  ak_skey_destroy( skey );
79  return error;
80  }
81  skey->data = NULL;
82  memset( &(skey->resource), 0, sizeof( union resource ));
83 
84  /* инициализируем генератор масок */
85  if(( error = ak_random_create_lcg( &skey->generator )) != ak_error_ok ) {
86  ak_error_message( error, __func__ , "wrong creation of random generator" );
87  ak_skey_destroy( skey );
88  return error;
89  }
90 
91  /* номер ключа генерится случайным образом. изменяется позднее, например,
92  при считывания с файлового носителя */
93  if(( error = ak_buffer_create_size( &skey->number,
94  ak_libakrypt_get_option( "key_number_length" ))) != ak_error_ok ) {
95  ak_error_message( error, __func__ ,"wrong creation key number buffer" );
96  ak_skey_destroy( skey );
97  return error;
98  }
99  if(( error = ak_skey_set_unique_number( skey )) != ak_error_ok ) {
100  ak_error_message( error, __func__ , "invalid creation of key number" );
101  ak_skey_destroy( skey );
102  return error;
103  }
104  /* OID ключа устанавливается производящей функцией */
105  skey->oid = NULL;
106 
107  /* После создания ключа все его флаги не определены */
108  skey->flags = 0;
109 
110  /* в заключение определяем указатели на методы.
111  по умолчанию используются механизмы для работы с аддитивной по модулю 2 маской.
112 
113  Указатели на функции, отличные от данных, должны устанавливаться при создании
114  конкретного типа ключа, например, конкретного алгоритма блочного шифрования */
115  skey->set_mask = ak_skey_set_mask_xor;
116  skey->remask = ak_skey_remask_xor;
117  skey->set_icode = ak_skey_set_icode_xor;
118  skey->check_icode = ak_skey_check_icode_xor;
119  skey->update = ak_skey_update_compress_xor;
120 
121  return ak_error_ok;
122 }

132 {
133  if( skey == NULL ) {
134  ak_error_message( ak_error_null_pointer, __func__ , "destroying a null pointer to secret key" );
135  return ak_error_null_pointer;
136  }
137 
138  /* удаляем данные */
139  ak_buffer_wipe( &(skey->key), &skey->generator );
140  ak_buffer_destroy( &(skey->key ));
141  memset( &(skey->key), 0, sizeof( struct buffer ));
142 
143  ak_buffer_wipe( &(skey->mask), &skey->generator );
144  ak_buffer_destroy( &(skey->mask ));
145  memset( &(skey->mask), 0, sizeof( struct buffer ));
146 
147  ak_buffer_wipe( &(skey->icode), &skey->generator );
148  ak_buffer_destroy( &(skey->icode ));
149  memset( &(skey->icode), 0, sizeof( struct buffer ));
150 
151  ak_random_destroy( &skey->generator );
152  if( skey->data != NULL )
153  /* удаляем память только при установленном флаге */
154  if( !((skey->flags)&ak_skey_flag_data_nonfree )) free( skey->data );
155 
156  ak_buffer_destroy( &skey->number );
157  skey->oid = NULL;
158  memset( &(skey->resource), 0, sizeof( union resource ));
159  skey->flags = 0;
160 
161  /* обнуляем указатели */
162  skey->set_mask = NULL;
163  skey->remask = NULL;
164  skey->set_icode = NULL;
165  skey->check_icode = NULL;
166  skey->update = NULL;
167 
168  return ak_error_ok;

179  int ak_skey_set_unique_number( ak_skey skey )
180 {
181  time_t tm = 0;
182  size_t len = 0;
183  struct hash ctx;
184  ak_uint8 out[32];
185  int error = ak_error_ok;
186  const char *version = ak_libakrypt_version();
187 
188  /* стандартные проверки */
189  if( skey == NULL ) return ak_error_message( ak_error_null_pointer,
190  __func__ , "using a null pointer to secret key" );
191  if(( error = ak_hash_create_streebog256( &ctx )) != ak_error_ok )
192  return ak_error_message( error, __func__ , "wrong creation of hash function context" );
193 
194  /* заполняем стандартное начало вектора */
195  memset( out, 0, 32 );
196  len = strlen( version );
197  memcpy( out, version, len ); /* сначала версия библиотеки */
198  tm = time( NULL );
199  memcpy( out+len, &tm, sizeof(tm) ); /* потом время генерации номера ключа */
200  len += sizeof( time_t );
201  if( len < 32 ) skey->generator.random( &skey->generator, out+len, 32 - len ); /* добавляем мусор */
202 
203  /* вычисляем номер и очищаем память */
204  ak_hash_context_ptr( &ctx, out, 32, out );
205  if(( error = ak_buffer_set_ptr( &skey->number,
206  out, ak_min( 32, skey->number.size ), ak_true )) != ak_error_ok )
207  return ak_error_message( ak_error_write_data, __func__ , "wrong assigning key number" );
208 
209  ak_hash_destroy( &ctx );
210  return ak_error_ok;
211 }

222  inline int ak_skey_check( ak_skey skey )
223 {
224  if( skey == NULL ) return ak_error_message( ak_error_null_pointer,
225  __func__ , "using a null pointer to secret key" );
226  /* проверяем длину ключа */
227  if( skey->key.data == NULL ) return ak_error_message( ak_error_null_pointer,
228  __func__ , "using a null pointer to key buffer" );
229  if( skey->key.size == 0 ) return ak_error_message( ak_error_zero_length, __func__ ,
230  "using a key buffer with zero length" );
231  if( skey->key.size%8 != 0 ) return ak_error_message( ak_error_wrong_length, __func__ ,
232  "using a key buffer with wrong length" );
233  /* проверяем маску */
234  if( skey->mask.data == NULL ) return ak_error_message( ak_error_null_pointer,
235  __func__ , "using a null pointer to key buffer" );
236  if( skey->mask.size != skey->key.size ) return ak_error_message( ak_error_wrong_length,
237  __func__ , "using mask and key buffer with diffenent sizes" );
238  /* проверяем контрольную сумму */
239  if( skey->icode.data == NULL ) return ak_error_message( ak_error_null_pointer,
240  __func__ , "using a null pointer to icode buffer" );
241  return ak_error_ok;
242 }

254  int ak_skey_set_mask_xor( ak_skey skey )
255 {
256  size_t idx = 0;
257  int error = ak_error_ok;
258 
259  /* выполняем стандартные проверки */
260  if(( error = ak_skey_check( skey )) != ak_error_ok )
261  return ak_error_message( error, __func__ , "using invalid secret key" );
262 
263  /* создаем маску*/
264  if(( error = skey->generator.random( &skey->generator,
265  skey->mask.data, skey->mask.size )) != ak_error_ok )
266  return ak_error_message( error, __func__ , "wrong mask generation for key buffer" );
267 
268  /* накладываем маску на ключ */
269  for( idx = 0; idx < ( skey->key.size >> 3 ); idx++ )
270  ((ak_uint64 *) skey->key.data)[idx] ^= ((ak_uint64 *) skey->mask.data)[idx];
271 
272  return error;
273 }

276  int ak_skey_update_compress_xor( ak_skey skey, ak_pointer ptr , const size_t size )
277 {
278  size_t idx = 0;
279  ak_uint8 tval[128];
280  int error = ak_error_ok;
281  ak_compress comp = ( ak_compress )ptr;
282 
283  if( skey == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
284  "using a null pointer to secret key" );
285  if( comp == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
286  "using a null pointer to compress context key" );
287  /* выполняем проверку длины ключа */
288  if( skey->key.size > 128 ) return ak_error_message( ak_error_undefined_value, __func__ ,
289  "using a very large secret key size" );
290  /* выполняем стандартные проверки */
291  if(( error = ak_skey_check( skey )) != ak_error_ok )
292  return ak_error_message( error, __func__ , "using invalid secret key" );
293 
294  /* снимаем маску */
295  for( idx = 0; idx < (skey->key.size >> 3); idx++ ) ((ak_uint64 *)tval)[idx] =
296  ((ak_uint64 *) skey->key.data)[idx] ^ ((ak_uint64 *) skey->mask.data)[idx];
297 
298  if(( error = ak_compress_update( comp, tval, ak_min( skey->key.size, size ))) != ak_error_ok )
299  ak_error_message( error, __func__ , "incorrect updating compress context" );
300  skey->generator.random( &skey->generator, tval, skey->key.size );
301 
302  return error;
303 }

316  int ak_skey_remask_xor( ak_skey skey )
317 {
318  size_t idx = 0;
319  ak_uint64 newmask = 0;
320  int error = ak_error_ok;
321 
322  /* выполняем стандартные проверки */
323  if(( error = ak_skey_check( skey )) != ak_error_ok )
324  return ak_error_message( error, __func__ , "using invalid secret key" );
325 
326  /* накладываем маску */
327  for( idx = 0; idx < ( skey->key.size >> 3 ); idx++ ) {
328  if( skey->generator.random( &skey->generator, &newmask, sizeof( ak_uint64 )) == ak_error_ok )
329  {
330  ((ak_uint64 *) skey->mask.data )[idx] ^= newmask;
331  ((ak_uint64 *) skey->key.data)[idx] ^= ((ak_uint64 *) skey->mask.data )[idx];
332  ((ak_uint64 *) skey->mask.data )[idx] = newmask;
333  } else return ak_error_message( ak_error_undefined_value,
334  __func__ , "wrong random mask generation" );
335  }
336 
337  /* удаляем старое */
338  memset( &newmask, 0, sizeof( ak_uint64 ));
339  return ak_error_ok;
340 }

363  static ak_uint64 ak_skey_icode_permutation( const ak_uint32 xv, const ak_uint32 yv )
364 {
365  ak_uint32 x = xv, y = yv, carry = 0;
366  ak_uint64 s = ( ak_uint64 )x + y, more = s&0x100000000, result = 0;
367 
368  if( more ) { x = ~x; y = ~y; s = ( ak_uint64 )x + y; }
369  result = y; result *= 3; result += x;
370  s *= s; result += s; if( result < s ) carry = 1;
371 
372  result >>= 1;
373  if( carry ) result ^= 0x8000000000000000L;
374  if( more ) result = ~result;
375  return result^0xC5BF891B4EF6AA79L; // константа есть \sqrt{\pi}
376 }

378 /* ----------------------------------------------------------------------------------------------- */
380  static void ak_skey_icode_xor_sum( ak_skey skey, ak_uint64 *result )
381 {
382  size_t i = 0;
383  for( i = 0; i < (skey->key.size >> 2); i+=4 ) {
384  ak_uint32 x = ((ak_uint32 *) skey->key.data)[i],
385  y = ((ak_uint32 *) skey->key.data)[i+2];
386  x ^= ((ak_uint32 *) skey->key.data)[i+1];
387  y ^= ((ak_uint32 *) skey->key.data)[i+3];
388  x ^= ((ak_uint32 *) skey->mask.data)[i];
389  x ^= ((ak_uint32 *) skey->mask.data)[i+1];
390  y ^= ((ak_uint32 *) skey->mask.data)[i+2];
391  y ^= ((ak_uint32 *) skey->mask.data)[i+3];
392  *result += ak_skey_icode_permutation( x, y );
393  }
394 }

492

494  if(( error = skey->set_mask( skey )) != ak_error_ok ) return ak_error_message( error,
495  __func__ , "wrong secret key masking" );
496 
497  if(( error = skey->set_icode( skey )) != ak_error_ok ) return ak_error_message( error,
498  __func__ , "wrong calculation of integrity code" );

502  skey->flags |= ak_skey_flag_set_key;
503

562  const ak_pointer salt, const size_t salt_size )
563 {
564  int error = ak_error_ok;
565 
566  /* выполняем необходимые проверки */
567  if( skey == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
568  "using a null pointer to secret key" );
569  if( skey->key.size == 0 ) return ak_error_message( ak_error_zero_length, __func__ ,
570  "using non initialized secret key context" );
571  if( pass == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
572  "using a null pointer to password" );
573  if( !pass_size ) return ak_error_message( ak_error_wrong_length, __func__ ,
574  "using a password with zero length" );
575  /* присваиваем буффер и маскируем его */
576  if(( error = ak_hmac_pbkdf2_streebog512( pass, pass_size, salt, salt_size,
577  ak_libakrypt_get_option("pbkdf2_iteration_count"),
578  skey->key.size, skey->key.data )) != ak_error_ok )
579  return ak_error_message( error, __func__ , "wrong generation a secret key data" );
580 
581  if(( error = skey->set_mask( skey )) != ak_error_ok ) return ak_error_message( error,
582  __func__ , "wrong secret key masking" );
583  if(( error = skey->set_icode( skey )) != ak_error_ok ) return ak_error_message( error,
584  __func__ , "wrong calculation of integrity code" );
585 
586  /* устанавливаем флаг того, что ключевое значение определено.
587  теперь ключ можно использовать в криптографических алгоритмах */
588  skey->flags |= ak_skey_flag_set_key;
589 
590  return error;
591 }
