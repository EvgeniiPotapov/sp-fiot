33  int ak_mac_create_oid( ak_mac mac, ak_oid oid )
34 {
35  int error = ak_error_ok;
36 
37  /* выполняем проверку */
38  if( mac == NULL ) return ak_error_message( ak_error_null_pointer, __func__,
39  "using null pointer to mac context" );
40  if( oid == NULL ) return ak_error_message( ak_error_null_pointer, __func__,
41  "using null pointer to mac function oid" );
42  /* проверяем, что oid от ключевой функции хеширования */
43  if( oid->engine != mac_function )
44  return ak_error_message( ak_error_oid_engine, __func__ , "using oid with wrong engine" );
45  /* проверяем, что oid от алгоритма, а не от параметров */
46  if( oid->mode != algorithm )
47  return ak_error_message( ak_error_oid_mode, __func__ , "using oid with wrong mode" );
48  /* проверяем, что производящая функция определена */
49  if( oid->func == NULL )
50  return ak_error_message( ak_error_undefined_function, __func__ ,
51  "using oid with undefined constructor function" );
52  /* инициализируем контекст */
53  if(( error = (( ak_function_mac_create *)oid->func)( mac )) != ak_error_ok )
54  return ak_error_message( error, __func__, "invalid creation of hash function context");
55 
56  return error;
57 }

64  int ak_mac_destroy( ak_mac mac )
65 {
66  int error = ak_error_ok;
67  if( mac == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
68  "destroying a null pointer to mac key context" );
69  switch( mac->type ) {
70  case type_hmac:
71  if(( error = ak_skey_destroy( &mac->choice._hmac.key )) != ak_error_ok )
72  ak_error_message( error, __func__ , "incorrect destroying secret key of mac context" );
73  if(( error = ak_hash_destroy( &mac->choice._hmac.ctx )) != ak_error_ok )
74  ak_error_message( error, __func__ , "incorrect destroying hash function of mac context" );
75  break;
76 
77  case type_imgost:
78  if(( error = ak_bckey_destroy( &mac->choice._imgost.bkey )) != ak_error_ok )
79  ak_error_message( error, __func__ , "incorrect destroying secret key of mac context" );
80  break;
81 
82  case type_mgm:
83  break;
84 
85  case type_signify:
86  break;
87 
88  default:
89  break;
90  }
91 
92  mac->type = type_undefined;
93  mac->bsize = 0;
94  mac->hsize = 0;
95  mac->clean = NULL;
96  mac->update = NULL;
97  mac->finalize = NULL;
98 
99  return error;
100 }

171  ak_buffer ak_mac_context_ptr( ak_mac mac, const ak_pointer in, const size_t size, ak_pointer out )
172 {
173  ak_skey skey = NULL;
174  int error = ak_error_ok;
175  size_t quot = 0, offset = 0;
176 
177  if( mac == NULL ) {
178  ak_error_message( ak_error_null_pointer, __func__ ,
179  "using null pointer to secret mac key context" );
180  return NULL;
181  }
182  if( in == NULL ) {
183  ak_error_message( ak_error_null_pointer, __func__ , "using a null pointer to input data" );
184  return NULL;
185  }
186 
187  if( mac->type == type_imgost ) {
188  return ak_bckey_context_mac_gost3413( &mac->choice._imgost.bkey, in, size, out );
189  }
190 
191  /* проверяем наличие ключа ( ресурс ключа проверяется при вызове clean ) */
192  if(( skey = ak_mac_context_get_secret_key( mac )) == NULL ) return NULL;
193  if( !((skey->flags)&ak_skey_flag_set_key )) {
194  ak_error_message( ak_error_key_value, __func__ , "using mac key with unassigned value" );
195  return NULL;
196  }
197 
198  /* вычищаем результаты предыдущих вычислений + инициализируем переменные */
199  if(( error = mac->clean( &mac->choice )) != ak_error_ok ) {
200  ak_error_message( error, __func__ , "incorrect cleaning of mac secret key context" );
201  return NULL;
202  }
203 
204  /* вычисляем фрагмент, длина которого кратна длине блока входных данных для хеш-функции */
205  quot = size/mac->bsize;
206  offset = quot*mac->bsize;
207  /* вызываем, если длина сообщения не менее одного полного блока */
208  if( quot > 0 )
209  if(( error = mac->update( &mac->choice, in, offset )) != ak_error_ok ) {
210  ak_error_message( error, __func__ , "wrong caclucation of mac update function" );
211  return NULL;
212  }
213 
214  /* обрабатываем хвост */
215  return mac->finalize( &mac->choice, (unsigned char *)in + offset, size - offset, out );
216 }

244

246  ak_error_message( error, __func__ , "wrong creation a compress context" );

249

251  ak_error_message( ak_error_undefined_function, __func__, "for imgost type this function not defined");
252  return NULL;

254

256  if(( error = ak_error_get_value( )) != ak_error_ok )

258 
259  ak_compress_destroy( &comp );
260  return result;

276  if( mac == NULL ) return ak_error_message( ak_error_null_pointer, __func__,

280 
281  if( mac->type == type_imgost ) {
282  if(( error = ak_bckey_context_set_ptr( &mac->choice._imgost.bkey,

284  return ak_error_message( error, __func__ , "incorrect assigning of key data" );
285  }
286 
287  /* присваиваем ключевой буффер */
288  if( mac->type == type_hmac ) {
289  if(( error = ak_skey_set_ptr( ak_mac_context_get_secret_key( mac ),
290  ptr, ak_min( size, mac->bsize ), ak_true )) != ak_error_ok )
291  return ak_error_message( error, __func__ , "incorrect assigning of key data" );
292  }
293 
294  return error;
295 }
