50  const char *description, ak_function_free_object *func )
51 {
52  int error = ak_error_ok;
53  ak_context_node node = NULL;
54 
55  /* минимальные проверки */
56  if( ctx == NULL ) {
57  ak_error_message( ak_error_null_pointer, __func__, "using a null pointer to context" );
58  return NULL;
59  }
60  /* для добавления ссылок на константы, нам пришлось убрать эту проверку
61  if( func == NULL ) {
62  ak_error_message( ak_error_undefined_function, __func__,
63  "using a null pointer to context free function" );
64  return NULL;
65  }
66  */
67  /* создаем контекст */
68  if(( node = malloc( sizeof( struct context_node ))) == NULL ) {
69  ak_error_message( ak_error_out_of_memory, __func__,
70  "wrong memory allocation for new context manager node" );
71  return NULL;
72  }
73 
74  /* инициализируем буффер и присваиваем ему значение */
75  if(( error = ak_buffer_create( &node->description )) != ak_error_ok ) {
76  ak_error_message( error, __func__, "wrong creation of internal buffer" );
77  return ( node = ak_context_node_delete( node ));
78  }
79 
80  if(( error = ak_buffer_set_str( &node->description, description )) != ak_error_ok ) {
81  ak_error_message( error, __func__, "wrong initialization of internal buffer" );
82  return ( node = ak_context_node_delete( node ));
83  }
84 
85  /* присваиваем остальные данные */
86  node->ctx = ctx;
87  node->id = id;
88  node->engine = engine;
89  node->free = func;
90  node->status = node_is_equal;
91

144 
145  if( manager == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
146  "using a null pointer to context manager structure" );

155  ak_error_message( error, __func__,
156  "wrong initialization a random generator from default crypto provider" );

158  if(( error = ak_random_create_lcg( &manager->key_generator )) != ak_error_ok )
159  return ak_error_message( error, __func__,
160  "wrong initialization of all types of random generators" );
161  }
162  #else
163  #error ak_context_manager_create(): using a non defined path of compilation
164  #endif
165 #endif

168  manager->size = ak_libakrypt_get_option("context_manager_size"); //ak_libakrypt_get_context_manager_size();
169 
170  if(( manager->array = malloc( manager->size*sizeof( ak_pointer ))) == NULL ) {
171  ak_context_manager_destroy( manager );
172  return ak_error_message( ak_error_out_of_memory, __func__ ,

195  int ak_context_manager_morealloc( ak_context_manager manager )
196 {
197  ak_context_node *newarray = NULL;
198  size_t idx = 0, newsize = (manager->size << 1);
199  int msize = ak_libakrypt_get_option("context_manager_max_size");
200 
201  if( manager == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
202  "using a null pointer to context manager structure" );
203  if( newsize <= manager->size )
204  return ak_error_message( ak_error_context_manager_size, __func__ ,
205  "unexpected value of new value of context manager's size" );
206  if(( msize < 0 ) || (( int )newsize > msize ))
207  return ak_error_message( ak_error_context_manager_max_size, __func__,
208  "current size of context manager exceeds permissible bounds" );
209 /* это отладочный вывод, который сообщает о выделении новой памяти
210  ak_error_message( ak_error_ok, __func__ ,
211  "TODO: load maximum of manager->size value from /etc/libakrypt.conf"); */
212 
213  if(( newarray = malloc( newsize*sizeof( ak_pointer ))) == NULL )
214  return ak_error_message( ak_error_out_of_memory, __func__ ,
215  "wrong memory allocation for context manager nodes" );
216  /* копируем данные и очищаем память */
217  for( idx = 0; idx < manager->size; idx++ ) {
218  newarray[idx] = manager->array[idx];
219  manager->array[idx] = NULL;
220  }
221  for( idx = manager->size; idx < newsize; idx++ ) newarray[idx] = NULL;
222 
223  free( manager->array );
224  manager->array = newarray;
225  manager->size = newsize;
226 
227  return ak_error_ok;
228 }

242 {
243  size_t idx = 0;
244  int error = ak_error_ok;
245 
246  if( manager == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
247  "using a null pointer to context manager structure" );
248  if( manager->array == NULL ) {
249  ak_error_message( error = ak_error_undefined_value, __func__ ,
250  "cleaning context manager with empty memory" );
251  } else {
252  /* удаляем ключевые структуры */
253  for( idx = 0; idx < manager->size; idx++ )
254  if( manager->array[idx] != NULL )
255  manager->array[idx] = ak_context_node_delete( manager->array[idx] );
256 
257  /* очищаем и уничтожаем память */
258  memset( manager->array, 0, manager->size);
259  free( manager->array );
260  manager->array = NULL;
261  }
262  manager->size = 0;
263 
264  /* удаляем генератор ключей */
265  if(( error = ak_random_destroy( &manager->key_generator )) != ak_error_ok )
266  return ak_error_message( error, __func__ , "using a null pointer to random key generator" );
267 
268  return error;
269 }

313  ak_handle ak_context_manager_add_node( ak_context_manager manager, ak_pointer ctx,
314  ak_oid_engine engine, const char *description, ak_function_free_object *func )
315 {
316  size_t idx = 0;
317  int error = ak_error_ok;
318  ak_context_node node = NULL;
319  ak_handle handle = ak_error_wrong_handle;
320 
321  /* минимальные проверки */
322  if( manager == NULL ) {
323  ak_error_message( ak_error_null_pointer, __func__, "using a null pointer to context manager" );
324  return ak_error_wrong_handle;
325  }
326  if( ctx == NULL ) {
327  ak_error_message( ak_error_null_pointer, __func__, "using a null pointer to context" );
328  return ak_error_wrong_handle;
329  }
330  if( description == NULL ) {
331  ak_error_message( ak_error_null_pointer, __func__,
332  "using a null pointer to description string" );
333  return ak_error_wrong_handle;
334  }
335 
336  /* блокируем доступ к структуре управления контекстами */
337  pthread_mutex_lock( &ak_context_manager_mutex );
338 
339  /* ищем свободный адрес */
340  for( idx = 0; idx < manager->size; idx++ ) {
341  if( manager->array[idx] == NULL ) break;
342  }
343  if( idx == manager->size ) {
344  if(( error = ak_context_manager_morealloc( manager )) != ak_error_ok ) {
345  ak_error_message( error, __func__, "wrong allocation a new memory for context manager" );
346  pthread_mutex_unlock( &ak_context_manager_mutex );
347  return ak_error_wrong_handle;
348  }
349  }
350 
351  /* адрес найден, теперь размещаем контекст */
352  handle = ak_context_manager_idx_to_handle( manager, idx );
353  if(( node = ak_context_node_new( ctx, handle, engine, description, func )) == NULL ) {
354  ak_error_message( ak_error_get_value(), __func__, "wrong creation of context manager node" );
355  pthread_mutex_unlock( &ak_context_manager_mutex );
356  return ak_error_wrong_handle;
357  }
358  manager->array[idx] = node;
359  pthread_mutex_unlock( &ak_context_manager_mutex );
360 
361  return handle;
362 }

393  int ak_context_manager_handle_check( ak_context_manager manager, ak_handle handle, size_t *idx )
394 {
395  /* проверяем менеджер контекстов */
396  if( manager == NULL ) return ak_error_message( ak_error_null_pointer, __func__,
397  "using null pointer to context manager" );
398  /* проверяем, что значение handle определено */
399  if( handle == ak_error_wrong_handle ) return ak_error_message( ak_error_wrong_handle,
400  __func__, "using an undefined handle value" );
401  /* определяем индекс */
402  *idx = ak_context_manager_handle_to_idx( manager, handle );
403 
404  /* проверяем границы */
405  if( *idx >= manager->size )
406  return ak_error_message( ak_error_wrong_handle, __func__, "invalid handle index" );
407 
408  /* проверяем наличие node */
409  if( manager->array[*idx] == NULL )
410  return ak_error_message( ak_error_null_pointer, __func__,
411  "using a null pointer to context manager node" );
412  /* проверяем наличие контекста */
413  if( manager->array[*idx]->ctx == NULL )
414  return ak_error_message( ak_error_null_pointer, __func__, "using null pointer to context" );
415 
416  return ak_error_ok;
417 }
