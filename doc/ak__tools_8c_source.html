75  char *ak_ptr_to_hexstr( const ak_pointer ptr, const size_t ptr_size, const ak_bool reverse )
76 {
77  char *nullstr = NULL;
78  size_t len = 1 + (ptr_size << 1);
79 
80  if( ptr == NULL ) {
81  ak_error_message( ak_error_null_pointer, __func__ , "using null pointer to data" );
82  return NULL;
83  }
84  if( ptr_size <= 0 ) {
85  ak_error_message( ak_error_zero_length, __func__ , "using data with zero or negative length" );
86  return NULL;
87  }
88 
89  if(( nullstr = (char *) malloc( len )) == NULL ) {
90  ak_error_message( ak_error_out_of_memory, __func__ , "incorrect memory allocation" );
91  }
92  else {
93  size_t idx = 0, js = 0, start = 0, offset = 2;
94  ak_uint8 *data = ( ak_uint8 * ) ptr;
95 
96  memset( nullstr, 0, len );
97  if( reverse ) { // движение в обратную сторону - от старшего байта к младшему
98  start = len-3; offset = -2;
99  }
100  for( idx = 0, js = start; idx < ptr_size; idx++, js += offset ) {
101  char str[4];
102  ak_snprintf( str, 3, "%02X", data[idx] );
103  memcpy( nullstr+js, str, 2 );
104  }
105  }
106  return nullstr;
107 }

214  int ak_hexstr_to_ptr( const char *hexstr, ak_pointer ptr, const size_t size, const ak_bool reverse )
215 {
216  int i = 0;
217  ak_uint8 *bdata = ptr;
218  size_t len = 0;
219 
220  if( hexstr == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
221  "using null pointer to a hex string" );
222  if( ptr == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
223  "using null pointer to a buffer" );
224  if( size == 0 ) return ak_error_message( ak_error_zero_length, __func__,
225  "using zero value for length of buffer" );
226  len = strlen( hexstr );
227  if( len&1 ) len++;
228  len >>= 1;
229  if( size < len ) return ak_error_message( ak_error_wrong_length, __func__ ,
230  "using a buffer with small length" );
231 
232  memset( ptr, 0, size ); // перед конвертацией мы обнуляем исходные данные
233  ak_error_set_value( ak_error_ok );
234  if( reverse ) {
235  for( i = strlen( hexstr )-2, len = 0; i >= 0 ; i -= 2, len++ ) {
236  bdata[len] = (ak_xconvert( hexstr[i] ) << 4) + ak_xconvert( hexstr[i+1] );
237  }
238  if( i == -1 ) bdata[len] = ak_xconvert( hexstr[0] );
239  } else {
240  for( i = 0, len = 0; i < (int) strlen( hexstr ); i += 2, len++ ) {
241  bdata[len] = (ak_xconvert( hexstr[i] ) << 4);
242  if( i < (int) strlen( hexstr )-1 ) bdata[len] += ak_xconvert( hexstr[i+1] );
243  }
244  }
245  return ak_error_get_value();
246 }

296  int ak_password_read( char *pass, const size_t psize )
297 {
298  size_t len = 0;
299  int error = ak_error_ok;
300 
301  #ifndef LIBAKRYPT_HAVE_TERMIOS_H
302  #ifdef _WIN32
303  char c = 0;
304  DWORD mode, count;
305  HANDLE ih = GetStdHandle( STD_INPUT_HANDLE );
306  if( !GetConsoleMode( ih, &mode ))
307  return ak_error_message( ak_error_terminal, __func__, "not connected to a console" );
308  SetConsoleMode( ih, mode & ~( ENABLE_ECHO_INPUT | ENABLE_LINE_INPUT ));
309 
310  memset( pass, 0, psize );
311  while( ReadConsoleA( ih, &c, 1, &count, NULL) && (c != '\r') && (c != '\n') && (len < psize-1) ) {
312  pass[len]=c;
313  len++;
314  }
315  pass[len]=0;
316 
317  /* восстанавливаем настройки консоли */
318  SetConsoleMode( ih, mode );
319  if(( len = strlen( pass )) < 1 )
320  return ak_error_message( ak_error_zero_length, __func__ , "input a very short password");
321  return error;
322 
323  #endif
324  return ak_error_undefined_function;
325 
326  #else
327  /* обрабатываем терминал */
328  struct termios ts, ots;
329 
330  tcgetattr( STDIN_FILENO, &ts); /* получаем настройки терминала */
331  ots = ts;
332  ts.c_cc[ VTIME ] = 0;
333  ts.c_cc[ VMIN ] = 1;
334  ts.c_iflag &= ~( BRKINT | INLCR | ISTRIP | IXOFF ); // ICRNL | IUTF8
335  ts.c_iflag |= IGNBRK;
336  ts.c_oflag &= ~( OPOST );
337  ts.c_cflag &= ~( CSIZE | PARENB);
338  ts.c_cflag |= CS8;
339  ts.c_lflag &= ~( ECHO | ICANON | IEXTEN | ISIG );
340  tcsetattr( STDIN_FILENO, TCSAFLUSH, &ts );
341  tcgetattr( STDIN_FILENO, &ts ); /* проверяем, что все установилось */
342  if( ts.c_lflag & ECHO ) {
343  ak_error_message( error = ak_error_terminal, __func__, "failed to turn off echo" );
344  goto lab_exit;
345  }
346 
347  memset( pass, 0, psize );
348  fgets( pass, psize, stdin );
349  if(( len = strlen( pass )) < 2 )
350  ak_error_message( error = ak_error_zero_length, __func__ , "input a very short password");
351  if( len > 0 ) pass[len-1] = 0;
352  else pass[0] = 0;
353 
354  /* убираем за собой и восстанавливаем настройки */
355  lab_exit: tcsetattr( STDIN_FILENO, TCSANOW, &ots );
356  return error;
357  #endif
358 
359  /* некорректный путь компиляции исходного текста функции */
360  return ak_error_undefined_function;
361 }

374  int ak_password_read_buffer( ak_buffer password )
375 {
376  int error = ak_error_ok;
377  if( password == NULL ) return ak_error_message( ak_error_null_pointer, __func__,
378  "using null pointer to password buffer" );
379  /* если нужно создаем буффер достаточно большого размера */
380  if((password->data == NULL ) || (password->size == 0 )) {
381  if(( error = ak_buffer_create_size( password, 256 )) != ak_error_ok )
382  return ak_error_message( error, __func__, "incorrect creation of fixed size buffer");
383  }
384  /* считываем пароль */
385  if(( error = ak_password_read( password->data, password->size )) != ak_error_ok ) {
386  return ak_error_message( error, __func__, "invalind password reading from standard console");
387  }
388  return ak_error_ok;
389 }
