250 /* ----------------------------------------------------------------------------------------------- */
251  ak_handle ak_libakrypt_find_oid_by_engine( ak_oid_engine engine )
252 {
253  size_t idx = 0;
254  ak_handle handle = ak_error_wrong_handle;
255 
256  if( engine != undefined_engine ) {
257  /* переборный цикл с первого элемента массива */
258  do{
259  if( libakrypt_oids[idx].engine == engine ) break; /* случай совпадения engine */
260  } while( ++idx < ak_libakrypt_oids_count( ));
261  if( idx == ak_libakrypt_oids_count( )) return handle;
262  }
263 
264  /* создаем дескриптор */
265  if(( handle = ak_context_manager_add_node( ak_libakrypt_get_context_manager(),
266  &libakrypt_oids[idx], oid_engine, "", NULL )) == ak_error_wrong_handle ) {
267  ak_error_message( ak_error_get_value(), __func__ , "wrong creation of context manager node" );
268  return ak_error_wrong_handle;
269  }
270 
271  return handle;
272 }

296 /* ----------------------------------------------------------------------------------------------- */
297  ak_handle ak_libakrypt_findnext_oid_by_engine( ak_handle handle, ak_oid_engine engine )
298 {
299  size_t idx = 0;
300  ak_oid oid = NULL;
301  int error = ak_error_ok;
302  ak_context_manager manager = ak_libakrypt_get_context_manager();
303 
304  /* для того чтобы не создавать новый дескриптор, приходится эмулировать
305  работу функции ak_handle_get_context() */
306  if( manager == NULL ) {
307  ak_error_message( ak_error_null_pointer, __func__, "null pointer to internal context manager");
308  return ak_error_wrong_handle;
309  }
310  if(( error = ak_context_manager_handle_check( manager, handle, &idx ))!= ak_error_ok ) {
311  ak_error_message( error, __func__, "wrong handle" );
312  return ak_error_wrong_handle;
313  }
314  if( manager->array[idx]->engine != oid_engine ) {
315  ak_error_message( ak_error_oid_engine, __func__, "using wrong engine for given handle" );
316  return ak_error_wrong_handle;
317  }
318 
319  /* получаем рабочий указатель на OID в качестве элемента глобального массива контекстов */
320  if(( oid = (ak_oid) manager->array[idx]->ctx ) == NULL ) {
321  ak_error_message( ak_error_null_pointer, __func__,
322  "using null pointer in handle to internal structure" );
323  return ak_error_wrong_handle;
324  }
325 
326  /* сдвигаемся по массиву OID вперед */
327  while( (++oid)->engine != undefined_engine ) {
328  /* в случае успешного поиска устанавливаем новое значение контекста в глобальном массиве */
329  if( engine == undefined_engine ) { manager->array[idx]->ctx = oid; return handle; }
330  if( oid->engine == engine ) { manager->array[idx]->ctx = oid; return handle; }
331  }
332  if(( error = ak_handle_delete( handle )) != ak_error_ok )
333  ak_error_message( error, __func__, "incorrect handle destroying");
334  return ak_error_wrong_handle;
335 }
