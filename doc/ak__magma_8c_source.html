113  static inline ak_uint32 ak_magma_gostf( ak_uint32 x )
114 {
115  /* очевидно, что можно хранить в памяти 32-х битные сдвинутые числа ! */
116 
117  x = ( (ak_uint32)( magma_boxes[3][(x>>24)&255] ) << 24 ) |
118  ( (ak_uint32)( magma_boxes[2][(x>>16)& 255] ) << 16 ) |
119  ( (ak_uint32)( magma_boxes[1][(x>>8)& 255] ) << 8 ) | ( (ak_uint32)( magma_boxes[0][x&255] ));
120 
121  /* заменить на один вызов циклического сдвига */
122  return x<<11 | x>>(32-11);
123 }

243  static int ak_skey_set_mask_additive( ak_skey skey )
244 {
245  size_t idx = 0;
246  int error = ak_error_ok;
247 
248  if( skey == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
249  "using a null pointer to secret key" );
250  /* выполняем стандартные проверки */
251  if(( error = ak_skey_check( skey )) != ak_error_ok )
252  return ak_error_message( error, __func__ , "using invalid secret key" );
253 
254  /* проверяем длину ключа */
255  if( skey->key.size != 32 ) return ak_error_message( ak_error_undefined_value, __func__ ,
256  "using a key buffer with wrong length" );
257  /* создаем маску*/
258  if( skey->generator.random( &skey->generator, skey->mask.data, skey->mask.size ) != ak_error_ok )
259  return ak_error_message( ak_error_write_data, __func__ ,
260  "wrong mask generation for key buffer" );
261  /* накладываем маску на ключ */
262  for( idx = 0; idx < (skey->key.size >> 2); idx++ )
263  ((ak_uint32 *) skey->key.data)[idx] += ((ak_uint32 *) skey->mask.data)[idx];
264 
265  return ak_error_ok;
266 }

269  static int ak_skey_update_hash_value_additive( ak_skey skey, ak_pointer ptr , const size_t size )
270 {
271  size_t idx = 0;
272  ak_uint8 tval[32];
273  int error = ak_error_ok;
274  ak_compress comp = ( ak_compress )ptr;
275 
276  if( skey == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
277  "using a null pointer to secret key" );
278  if( comp == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
279  "using a null pointer to compress context key" );
280  /* выполняем стандартные проверки */
281  if(( error = ak_skey_check( skey )) != ak_error_ok )
282  return ak_error_message( error, __func__ , "using invalid secret key" );
283 
284  /* проверяем длину ключа */
285  if( skey->key.size != 32 ) return ak_error_message( ak_error_undefined_value, __func__ ,
286  "using a key buffer with wrong length" );
287 
288  /* снимаем маску */
289  for( idx = 0; idx < (skey->key.size >> 2); idx++ ) ((ak_uint32 *)tval)[idx] =
290  ((ak_uint32 *) skey->key.data)[idx] - ((ak_uint32 *) skey->mask.data)[idx];
291  if(( error = ak_compress_update( comp, tval, ak_min( 32, size ))) != ak_error_ok )
292  ak_error_message( error, __func__ , "incorrect updating compress context" );
293  skey->generator.random( &skey->generator, tval, 32 );
294 
295  return error;
296 }

361  static ak_uint64 ak_skey_icode_permutation( const ak_uint32 xv, const ak_uint32 yv )
362 {
363  ak_uint32 x = xv, y = yv, carry = 0;
364  ak_uint64 s = ( ak_uint64 )x + y, more = s&0x100000000, result = 0;
365 
366  if( more ) { x = ~x; y = ~y; s = ( ak_uint64 )x + y; }
367  result = y; result *= 3; result += x;
368  s *= s; result += s; if( result < s ) carry = 1;
369 
370  result >>= 1;
371  if( carry ) result ^= 0x8000000000000000L;
372  if( more ) result = ~result;
373  return result^0xC5BF891B4EF6AA79L; // константа есть \sqrt{\pi}
374 }

379 {
380  size_t i = 0;
381  for( i = 0; i < (skey->key.size >> 2); i+=4 ) {
382  ak_uint32 x = ((ak_uint32 *) skey->key.data)[i],
383  y = ((ak_uint32 *) skey->key.data)[i+2];
384  x += ((ak_uint32 *) skey->key.data)[i+1];
385  y += ((ak_uint32 *) skey->key.data)[i+3];
386  x -= ((ak_uint32 *) skey->mask.data)[i];

388  y -= ((ak_uint32 *) skey->mask.data)[i+2];
389  y -= ((ak_uint32 *) skey->mask.data)[i+3];
390  *result += ak_skey_icode_permutation( x, y );
391  }
