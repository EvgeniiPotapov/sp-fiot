40 #ifndef LIBAKRYPT_HAVE_ASM_CODE
41  /* реализация метода Карацубы для двух 64-х битных чисел */
42  #define umul_ppmm( w1, w0, u, v ) \
43  do { \
44  ak_uint64 __x0, __x1, __x2, __x3; \
45  ak_uint32 __ul, __vl, __uh, __vh; \
46  ak_uint64 __u = (u), __v = (v); \
47  \
48  __ul = __u & 0xFFFFFFFF; \
49  __uh = __u >> 32; \
50  __vl = __v & 0xFFFFFFFF; \
51  __vh = __v >> 32; \
52  \
53  __x0 = (ak_uint64) __ul * __vl; \
54  __x1 = (ak_uint64) __ul * __vh; \
55  __x2 = (ak_uint64) __uh * __vl; \
56  __x3 = (ak_uint64) __uh * __vh; \
57  \
58  __x1 += ( __x0 >> 32 ); \
59  __x1 += __x2; \
60  if (__x1 < __x2) __x3 += ((ak_uint64)1 << 32 ); \
61  \
62  (w1) = __x3 + (__x1 >> 32 ); \
63  (w0) = ( __x1 << 32 ) + ( __x0 & 0xFFFFFFFF ); \
64  } while (0)
65 #endif

426  void ak_mpzn_rem( ak_uint64 *r, ak_uint64 *u, ak_uint64 *p, const size_t size )
427 {
428  ak_uint64 q = 0;
429  ak_mpznmax z, s;
430 
431  if( p[size-1] != -1 ) q = u[size-1]/(1+p[size-1]);
432 
433  /* проверяем, нужно ли приведение, или же вычет меньше модуля */
434  if( q == 0 ) {
435  if( r != u ) memcpy( r, u, size*sizeof( ak_uint64 ));
436  return;
437  }
438  /* выполняем умножение и последующее вычитание */
439  if( q > 1 ) {
440  ak_mpzn_mul_ui( s, p, size, q );
441  ak_mpzn_sub( z, u, s, size );
442  } else ak_mpzn_sub( z, u, p, size );
443 
444  if( ak_mpzn_sub( s, z, p, size )) memcpy( r, z, size*sizeof( ak_uint64 ));
445  else memcpy( r, s, size*sizeof( ak_uint64 ));
446 }

465  inline void ak_mpzn_add_montgomery( ak_uint64 *z, ak_uint64 *x, ak_uint64 *y,
466  ak_uint64 *p, const size_t size )
467 {
468  size_t i = 0;
469  ak_uint64 av = 0, bv = 0, cy = 0;
470  ak_mpznmax t = ak_mpznmax_zero;
471 
472  // сначала складываем: (x + y) -> t
473  for( i = 0; i < size; i++, x++, y++ ) {
474  av = *x; bv = *y;
475  bv += cy;
476  cy = bv < cy;
477  bv += av;
478  cy += bv < av;
479  t[i] = bv;
480  }
481  t[size] = cy; cy = 0;
482  // потом вычитаем: (t - p) -> z
483  for( i = 0; i < size; i++, p++ ) {
484  av = t[i];
485  bv = av - cy;
486  cy = bv > av;
487  av = bv - *p;
488  cy += av > bv;
489  z[i] = av;
490  }
491  if( t[size] != cy ) memcpy( z, t, size*sizeof( ak_uint64 ));
492 }

508  void ak_mpzn_lshift_montgomery( ak_uint64 *z, ak_uint64 *x, ak_uint64 *p, const size_t size )
509 {
510  size_t i;
511  ak_uint64 av = 0, bv = 0, cy = 0;
512  ak_mpznmax t = ak_mpznmax_zero;
513 
514  t[size] = 0;
515  for( i = 0; i < size; i++, p++ ) {
516  t[i+1] = (( x[i]&0x8000000000000000LL ) > 0 );
517  t[i] |= x[i] << 1; // сначала сдвигаем на один разряд влево
518  av = t[i]; // потом вычитаем модуль
519  bv = av - cy;
520  cy = bv > av;

522  cy += av > bv;
523  z[i] = av;
524  }

546  ak_uint64 *p, ak_uint64 n0, const size_t size )
547 {

574  ak_uint64 w1, w0, cy;

620  void ak_mpzn_modpow_montgomery( ak_uint64 *z, ak_uint64 *x, ak_uint64 *k,
621  ak_uint64 *p, ak_uint64 n0, const size_t size )
622 {
623  ak_uint64 uk = 0;
624  size_t s = size-1;
625  long long int i, j;
626  ak_mpznmax res = ak_mpznmax_zero; // это константа r (mod p) = r-p
627  if( ak_mpzn_sub( res, res, p, size ) == 0 ) {
628  ak_error_message( ak_error_undefined_value,
629  "using an unexpected value of prime modulo", __func__ );
630  return;
631  }
632  while( k[s] == 0 ) {
633  if( s > 0 ) --s;
634  else {
635  ak_mpzn_set( z, res, size );
636  return;
637  }
638  }
639  for( i = s; i >= 0; i-- ) {
640  uk = k[i];
641  for( j = 0; j < 64; j++ ) {
642  ak_mpzn_mul_montgomery( res, res, res, p, n0, size );
643  if( uk&0x8000000000000000LL ) ak_mpzn_mul_montgomery( res, res, x, p, n0, size );
644  uk <<= 1;
645  }
646  }
647  memcpy( z, res, size*sizeof( ak_uint64 ));
648 }
