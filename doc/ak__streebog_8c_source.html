44 /* ----------------------------------------------------------------------------------------------- */
46  static inline void streebog_lps( ak_uint64 *result, const ak_uint64 *data )
47 {
48  int idx = 0, idx2 = 0;
49  unsigned char *a = (unsigned char *)data; /* приводим к массиву байт */
50 
51  /* Все три преобразования вместе */
52  /* (этот очень короткий код был предложен Павлом Лебедевым */
53  for( idx = 0; idx < 8; idx++ ) {
54  ak_uint64 sidx = idx, c = 0;
55  for( idx2 = 0; idx2 < 8; idx2++, sidx += 8 )
56  c ^= streebog_Areverse_expand[idx2][gost_pi[a[sidx]]];
57  result[idx] = c;
58  }
59 }
60 
61 /* ----------------------------------------------------------------------------------------------- */
63  static inline void streebog_x( ak_uint64 *r, const ak_uint64 *k, const ak_uint64 *a )
64 {
65  int idx = 0;
66  for( idx = 0; idx < 8; idx++ ) r[idx] = k[idx] ^ a[idx];
67 }
68 
69 /* ----------------------------------------------------------------------------------------------- */
71  static inline void streebog_g( struct streebog *ctx, ak_uint64 *n, const ak_uint64 *m )
72 {
73  int idx = 0;
74  ak_uint64 K[8], T[8], B[8];
75  if( n != NULL ) {
76  streebog_x( B, ctx->H, n );
77  streebog_lps( K, B );
78  }
79  else
80  streebog_lps( K, ctx->H );
81  /* K - ключ K1 */
82  for( idx = 0; idx < 8; idx++ ) T[idx] = m[idx]; /* memcpy( T, m, 64 ); */
83  for( idx = 0; idx < 12; idx++ ) {
84  streebog_x( B, T, K ); streebog_lps( T, B ); /* преобразуем текст */
85  streebog_x( B, K, streebog_c[idx] ); streebog_lps( K, B ); /* новый ключ */
86  }
87  /* изменяем значение переменной h */
88  for ( idx = 0; idx < 8; idx++ ) ctx->H[idx] ^= T[idx] ^ K[idx] ^ m[idx];
89 }

91 /* ----------------------------------------------------------------------------------------------- */
93  static inline void streebog_add( struct streebog *ctx, ak_uint64 size )
94 {
95  ak_uint64 tmp = size + ctx->N[0];
96  if( tmp < ctx->N[0] ) ctx->N[1]++;
97  ctx->N[0] = tmp; /* такой код позволяет обработать сообщения, длиной не более 2^125 байт */
98 }
99 
100 /* ----------------------------------------------------------------------------------------------- */
102  static inline void streebog_sadd( struct streebog *ctx, const ak_uint64 *data )
103 {
104  int i = 0;
105  ak_uint64 carry = 0;
106  for( i = 0; i < 8; i++ )
107  {
108  if( carry ) {
109  ctx->SIGMA[i] ++;
110  if( ctx->SIGMA[i] ) carry = 0;
111  }
112  ctx->SIGMA[i] += data[i];
113  if( ctx->SIGMA[i] < data[i] ) carry = 1;
114  }
115 }

121  static int ak_hash_streebog_clean( ak_pointer ctx )
122 {
123  struct streebog *sx = NULL;
124  if( ctx == NULL ) return ak_error_message( ak_error_null_pointer,
125  __func__ , "using null pointer to a context" );
126 
127  sx = ( struct streebog * ) (( ak_hash ) ctx )->data;
128  memset( sx->N, 0, 64 );
129  memset( sx->SIGMA, 0, 64 );
130  if( (( ak_hash ) ctx )->hsize == 32 ) memset( sx->H, 1, 64 );
131  else memset( sx->H, 0, 64 );
132 
133  return ak_error_ok;
134 }

136 /* ----------------------------------------------------------------------------------------------- */
138  static int ak_hash_streebog_update( ak_pointer ctx, const ak_pointer in, const size_t size )
139 {
140  ak_uint64 quot = 0, *dt = NULL;
141  struct streebog *sx = NULL;
142 
143  if( ctx == NULL ) return ak_error_message( ak_error_null_pointer,
144  __func__ , "using null pointer to a context" );
145  if( !size ) return ak_error_message( ak_error_zero_length,
146  __func__ , "using zero length for hash data" );
147  quot = size/(( ak_hash ) ctx )->bsize;
148  if( size - quot*(( ak_hash ) ctx )->bsize ) /* длина данных должна быть кратна ctx->bsize */
149  return ak_error_message( ak_error_wrong_length, __func__ , "using data with wrong length" );
150 
151  dt = ( ak_uint64 *) in;
152  sx = ( struct streebog * ) (( ak_hash ) ctx )->data;
153  do{
154  streebog_g( sx, sx->N, dt );
155  streebog_add( sx, 512 );
156  streebog_sadd( sx, dt );
157  quot--; dt += 8;
158  } while( quot > 0 );
159 
160  return ak_error_ok;
161 }

163 /* ----------------------------------------------------------------------------------------------- */
164  static ak_buffer ak_hash_streebog_finalize( ak_pointer ctx, const ak_pointer in,
165  const size_t size, ak_pointer out )
166 {
167  ak_uint64 m[8];
168  ak_pointer pout = NULL;
169  ak_buffer result = NULL;
170  unsigned char *mhide = NULL;
171  struct streebog sx; /* структура для хранения копии текущего состояния контекста */
172 
173  if( ctx == NULL ) {
174  ak_error_message( ak_error_null_pointer, __func__ , "using null pointer to a context" );
175  return NULL;
176  }
177  if( size >= 64 ) {
178  ak_error_message( ak_error_zero_length, __func__ ,
179  "using wrong length for finalized hash data" );
180  return NULL;
181  }
182 
183  /* формируем временный текст */
184  memset( m, 0, 64 );
185  if( in != NULL )
186  memcpy( m, in, ( ak_uint32 )size ); // здесь приведение типов корректно, поскольку 0 <= size < 64
187  mhide = ( unsigned char * )m;
188  mhide[size] = 1; /* дополнение */
189 
190  /* при финализации мы изменяем копию существующей структуры */
191  memcpy( &sx, ( struct streebog * ) (( ak_hash ) ctx )->data, sizeof( struct streebog ));
192  streebog_g( &sx, sx.N, m );
193  streebog_add( &sx, size << 3 );
194  streebog_sadd( &sx, m );
195  streebog_g( &sx, NULL, sx.N );
196  streebog_g( &sx, NULL, sx.SIGMA );
197 
198  /* определяем указатель на область памяти, в которую будет помещен результат вычислений */
199  if( out != NULL ) pout = out;
200  else {
201  if(( result = ak_buffer_new_size((( ak_hash )ctx)->hsize )) != NULL ) pout = result->data;
202  else ak_error_message( ak_error_get_value( ), __func__ ,
203  "wrong creation of result buffer" );
204  }
205 
206  /* копируем нужную часть результирующего массива или выдаем сообщение об ошибке */
207  if( pout != NULL ) {
208  if((( ak_hash )ctx)->hsize == 64 ) memcpy( pout, sx.H, 64 );
209  else memcpy( pout, sx.H+4, 32 );
210  } else ak_error_message( ak_error_out_of_memory, __func__ ,
211  "incorrect memory allocation for result buffer" );
212  return result;
213 }

222  int ak_hash_create_streebog256( ak_hash ctx )
223 {
224  int error = ak_error_ok;
225 
226  /* выполняем проверку */
227  if( ctx == NULL ) return ak_error_message( ak_error_null_pointer, __func__,
228  "using null pointer to hash context" );
229  /* инициализируем контекст */
230  if(( error = ak_hash_create( ctx, sizeof( struct streebog ), 64 )) != ak_error_ok )
231  return ak_error_message( error, __func__ , "incorrect streebog context creation" );
232 
233  /* устанавливаем размер хешхода и OID алгоритма хеширования */
234  ctx->hsize = 32; /* длина хешкода составляет 256 бит */
235  if(( ctx->oid = ak_oid_find_by_name( "streebog256" )) == NULL )
236  return ak_error_message( ak_error_get_value(), __func__, "internal OID search error");
237 
238  /* устанавливаем функции - обработчики событий */
239  ctx->clean = ak_hash_streebog_clean;
240  ctx->update = ak_hash_streebog_update;
241  ctx->finalize = ak_hash_streebog_finalize;
242 
243  /* инициализируем память */
244  ak_hash_streebog_clean( ctx );
245  return error;
246 }
