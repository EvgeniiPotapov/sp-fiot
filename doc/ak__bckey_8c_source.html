59  int ak_bckey_create( ak_bckey bkey, size_t keysize, size_t blocksize )
60 {
61  int error = ak_error_ok;
62  if( bkey == NULL ) return ak_error_message( ak_error_null_pointer, __func__,
63  "using a null pointer to block cipher context" );
64  if( !keysize ) return ak_error_message( ak_error_zero_length, __func__,
65  "using block cipher key with zero length" );
66  if( !blocksize ) return ak_error_message( ak_error_zero_length, __func__,
67  "using cipher with zero block length" );
68  /* теперь инициализируем данные,
69  для ключей блочного шифрования длина контрольной суммы всегда равна 8 байт */
70  if(( error = ak_skey_create( &bkey->key, keysize, 8 )) != ak_error_ok )
71  return ak_error_message( error, __func__, "wrong creation of secret key" );
72 
73  /* длина инициализационного вектора всегда совпадает с длиной блока данных */
74  if(( error = ak_buffer_create_size( &bkey->ivector, blocksize )) != ak_error_ok ) {
75  if( ak_skey_destroy( &bkey->key ) != ak_error_ok )
76  ak_error_message( ak_error_get_value(), __func__, "wrong destroying a secret key" );
77  return ak_error_message( error, __func__, "wrong memory allocation for temporary vector");
78  }
79  bkey->encrypt = NULL;
80  bkey->decrypt = NULL;
81  bkey->schedule_keys = NULL;
82  bkey->delete_keys = NULL;
83 
84  return ak_error_ok;
85 }

93 {
94  int error = ak_error_ok;
95  if( bkey == NULL ) return ak_error_message( ak_error_null_pointer, __func__,
96  "using a null pointer to block cipher context" );
97  if( bkey->delete_keys != NULL ) {
98  if(( error = bkey->delete_keys( &bkey->key )) != ak_error_ok ) {
99  ak_error_message( error, __func__ , "wrong deleting of round keys" );
100  }
101  }
102  if(( error = ak_buffer_wipe( &bkey->ivector, &bkey->key.generator )) != ak_error_ok )
103  ak_error_message( error, __func__, "wrong wiping a temporary vector");
104  if(( error = ak_buffer_destroy( &bkey->ivector )) != ak_error_ok )
105  ak_error_message( error, __func__, "wrong destroying a temporary vector" );
106  if(( error = ak_skey_destroy( &bkey->key )) != ak_error_ok )
107  ak_error_message( error, __func__, "wrong destroying a secret key" );
108 
109  bkey->encrypt = NULL;
110  bkey->decrypt = NULL;
111  bkey->schedule_keys = NULL;
112  bkey->delete_keys = NULL;
113 
114  return error;

265  int ak_bckey_context_encrypt_ecb( ak_bckey bkey, ak_pointer in, ak_pointer out, size_t size )
266 {
267  ak_int64 blocks = 0;
268  ak_uint64 *inptr = (ak_uint64 *)in, *outptr = (ak_uint64 *)out;
269 
270  /* выполняем проверку размера входных данных */
271  if( size%bkey->ivector.size != 0 )
272  return ak_error_message( ak_error_wrong_block_cipher_length,
273  __func__ , "the length of input data is not divided by block length" );
274 
275  /* проверяем целостность ключа */
276  if( bkey->key.check_icode( &bkey->key ) != ak_true )
277  return ak_error_message( ak_error_wrong_key_icode,
278  __func__, "incorrect integrity code of secret key value" );
279  /* уменьшаем значение ресурса ключа */
280  blocks = (ak_uint64 ) size/bkey->ivector.size;
281  if( bkey->key.resource.counter < blocks )
282  return ak_error_message( ak_error_low_key_resource,
283  __func__ , "low resource of block cipher key" );
284  else bkey->key.resource.counter -= blocks;
285 
286  /* теперь приступаем к зашифрованию данных */
287  if( bkey->ivector.size == 8 ) { /* здесь длина блока равна 64 бита */
288  do {
289  bkey->encrypt( &bkey->key, inptr++, outptr++ );
290  } while( --blocks > 0 );
291  }
292  if( bkey->ivector.size == 16 ) { /* здесь длина блока равна 128 бит */
293  do {
294  bkey->encrypt( &bkey->key, inptr, outptr );
295  inptr+=2; outptr+=2;
296  } while( --blocks > 0 );
297  }
298 
299  /* перемаскируем ключ */
300  if( bkey->key.remask( &bkey->key ) != ak_error_ok )
301  ak_error_message( ak_error_get_value(), __func__ , "wrong remasking of secret key" );
302 
303  return ak_error_ok;
304 }

379  int ak_bckey_context_xcrypt( ak_bckey bkey, ak_pointer in, ak_pointer out, size_t size,
380  ak_pointer iv, size_t iv_size )
381 {
382  ak_int64 blocks = (ak_int64)size/bkey->ivector.size,
383  tail = (ak_int64)size%bkey->ivector.size;
384  ak_uint64 yaout[2], *inptr = (ak_uint64 *)in, *outptr = (ak_uint64 *)out;
385 
386  /* проверяем целостность ключа */
387  if( bkey->key.check_icode( &bkey->key ) != ak_true )
388  return ak_error_message( ak_error_wrong_key_icode, __func__,
389  "incorrect integrity code of secret key value" );
390  /* проверяем длину синхропосылки (если меньше половины блока, то плохо)
391  если больше - то лишнее не используется */
392  if( iv_size < ( bkey->ivector.size >> 1 ))
393  return ak_error_message( ak_error_wrong_iv_length, __func__,
394  "incorrect length of initial value" );
395  /* уменьшаем значение ресурса ключа */
396  if( bkey->key.resource.counter < ( blocks + ( tail > 0 )))
397  return ak_error_message( ak_error_low_key_resource,
398  __func__ , "low resource of block cipher key" );
399  else bkey->key.resource.counter -= ( blocks + ( tail > 0 )); /* уменьшаем ресурс ключа */
400 
401  /* теперь приступаем к зашифрованию данных */
402  if( bkey->key.flags&ak_flag_xcrypt_update ) bkey->key.flags ^= ak_flag_xcrypt_update;
403  memset( bkey->ivector.data, 0, bkey->ivector.size );
404 
405  if( blocks ) {
406  /* здесь длина блока равна 64 бита */
407  if( bkey->ivector.size == 8 ) {
408  memcpy( ((ak_uint8 *)bkey->ivector.data)+4, iv, 4 );
409  do {
410  bkey->encrypt( &bkey->key, bkey->ivector.data, yaout );
411  *outptr = *inptr ^ yaout[0];
412  outptr++; inptr++; ((ak_uint64 *)bkey->ivector.data)[0]++;
413  } while( --blocks > 0 );
414  }
415 
416  /* здесь длина блока равна 128 бит */
417  if( bkey->ivector.size == 16 ) {
418  memcpy( ((ak_uint8 *)bkey->ivector.data)+8, iv, 8 );
419  do {
420  bkey->encrypt( &bkey->key, bkey->ivector.data, yaout );
421  *outptr = *inptr ^ yaout[0]; outptr++; inptr++;
422  *outptr = *inptr ^ yaout[1]; outptr++; inptr++;
423  ((ak_uint64 *)bkey->ivector.data)[0]++; // здесь мы не учитываем знак переноса
424  // потому что объем данных на одном ключе не должен превышать
425  // 2^64 блоков (контролируется через ресурс ключа)
426  } while( --blocks > 0 );
427  }
428  }
429 
430  if( tail ) { /* на последок, мы обрабатываем хвост сообщения */
431  size_t i;
432  bkey->encrypt( &bkey->key, bkey->ivector.data, yaout );
433  for( i = 0; i < tail; i++ ) /* теперь мы гаммируем tail байт, используя для этого
434  старшие байты (most significant bytes) зашифрованного счетсчикаы */
435  ( (ak_uint8*)outptr )[i] =
436  ( (ak_uint8*)inptr )[i]^( (ak_uint8 *)yaout)[bkey->ivector.size-tail+i];
437  /* запрещаем дальнейшее использование xcrypt_update для данных, длина которых не кратна длине блока */
438  memset( bkey->ivector.data, 0, bkey->ivector.size );
439  bkey->key.flags |= ak_flag_xcrypt_update;
440  }
441 
442  /* перемаскируем ключ */
443  if( bkey->key.remask( &bkey->key ) != ak_error_ok )
444  return ak_error_message( ak_error_get_value(), __func__ , "wrong remasking of secret key" );
445 
446  return ak_error_ok;
447 }

552  /* проверяем целостность ключа */
553  if( bkey->key.check_icode( &bkey->key ) != ak_true ) {
554  ak_error_message( ak_error_wrong_key_icode, __func__,
555  "incorrect integrity code of secret key value" );
556  return NULL;
557  }
558 
559  /* уменьшаем значение ресурса ключа */
560  if( bkey->key.resource.counter < ( blocks + ( tail > 0 ))) {
561  ak_error_message( ak_error_low_key_resource, __func__ , "low resource of block cipher key" );
562  return NULL;
563  } else bkey->key.resource.counter -= ( blocks + ( tail > 0 )); /* уменьшаем ресурс ключа */
564 
565  memset( akey, 0, sizeof( akey ));
566  memset( yaout, 0, sizeof( yaout ));
567  if( !tail ) { tail = bkey->ivector.size; blocks--; } /* последний блок всегда существует */
568 
569  /* основной цикл */
570  if( bkey->ivector.size == 16 ) { /* здесь длина блока равна 128 бита */
571  for( i = 0; i < blocks; i++, inptr += 2 ) {
572  yaout[0] ^= inptr[0];
573  yaout[1] ^= inptr[1];
574  bkey->encrypt( &bkey->key, yaout, yaout );
575  }
576 
577  bkey->encrypt( &bkey->key, akey, akey );
578  if( akey[1]&0x8000000000000000LL ) {
579  akey[1] <<= 1; akey[1] ^= (akey[0] >> 63); akey[0] <<= 1;
580  akey[0] ^= 0x87; //135
581  } else {
582  akey[1] <<= 1; akey[1] ^= (akey[0] >> 63); akey[0] <<= 1;
583  }
584 
585  if( tail < bkey->ivector.size ) {
586  if( akey[1]&0x8000000000000000LL ) {
587  akey[1] <<= 1; akey[1] ^= (akey[0] >> 63); akey[0] <<= 1;
588  akey[0] ^= 0x87; //135
589  } else {
590  akey[1] <<= 1; akey[1] ^= (akey[0] >> 63); akey[0] <<= 1;
591  }
592  ((ak_uint8 *)akey)[tail] ^= 0x80;
593  }
594 
595  // теперь шифруем последний блок
596  akey[0] ^= yaout[0]; akey[1] ^= yaout[1];
597  for( i = 0; i < tail; i++ ) ((ak_uint8 *)akey)[i] ^= ((ak_uint8 *)inptr)[i];
598  bkey->encrypt( &bkey->key, akey, akey );
599  }
600 
601  if( bkey->ivector.size == 8 ) { /* здесь длина блока равна 64 бита */
602  for( i = 0; i < blocks; i++, inptr++ ) {
603  yaout[0] ^= inptr[0];
604  bkey->encrypt( &bkey->key, yaout, yaout );
605  }
606 
607  bkey->encrypt( &bkey->key, akey, akey );
608  if( akey[0]&0x8000000000000000LL ) {
609  akey[0] <<= 1; akey[0] ^= 0x1B;
610  } else akey[0] <<= 1;
611 
612  if( tail < bkey->ivector.size ) {
613  if( akey[0]&0x8000000000000000LL ) {
614  akey[0] <<= 1; akey[0] ^= 0x1B;
615  } else akey[0] <<= 1;
616  ((ak_uint8 *)akey)[tail] ^= 0x80;
617  }
618 
619  // теперь шифруем последний блок
620  akey[0] ^= yaout[0];
621  for( i = 0; i < tail; i++ ) ((ak_uint8 *)akey)[i] ^= ((ak_uint8 *)inptr)[i];
622  bkey->encrypt( &bkey->key, akey, akey );
623  }
624 
625  /* определяем указатель на область памяти, в которую будет помещен результат вычислений */
626  if( out != NULL ) pout = out;
627  else {
628  if(( result = ak_buffer_new_size( bkey->ivector.size )) != NULL ) pout = result->data;
629  else ak_error_message( ak_error_get_value( ), __func__ , "wrong creation of result buffer" );
630  }
631  /* копируем нужную часть результирующего массива или выдаем сообщение об ошибке */
632  if( pout != NULL ) memcpy( pout, akey, bkey->ivector.size );
633  else ak_error_message( ak_error_out_of_memory, __func__ ,
634  "incorrect memory allocation for result buffer" );
635  return result;
636 }
