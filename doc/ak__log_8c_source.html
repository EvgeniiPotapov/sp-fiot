113 {
114  pthread_mutex_lock( &ak_function_log_default_mutex );
115  if( function != NULL ) ak_function_log_default = function;

133 /* ----------------------------------------------------------------------------------------------- */
134  int ak_log_set_message( const char *message )
135 {
136  int result = 0;
137  if( ak_function_log_default == NULL ) return ak_error_set_value( ak_error_undefined_function );
138  if( message == NULL ) {
139  return ak_error_message( ak_error_null_pointer, __func__ , "using a NULL string for message" );
140  } else {
141  pthread_mutex_lock( &ak_function_log_default_mutex );
142  result = ak_function_log_default( message );
143  pthread_mutex_unlock( &ak_function_log_default_mutex );
144  return result;
145  }
146  return ak_error_ok;
147 }

186  int ak_error_message( const int code, const char *function, const char *message )
187 {
188  /* здесь мы выводим в логгер строку вида [pid] function: message (code: n) */
189  char error_event_string[1024];
190  memset( error_event_string, 0, 1024 );
191 
192 #ifdef LIBAKRYPT_HAVE_UNISTD_H
193  ak_snprintf( error_event_string, 1023, "[%d] %s(): %s (code: %d)",
194  getpid(), function, message, code );
195 #else
196  #ifdef _MSC_VER
197  ak_snprintf( error_event_string, 1023, "[%d] %s(): %s (code: %d)",
198  GetCurrentProcessId(), function, message, code );
199  #else
200  #error Unsupported path to compile, sorry ...
201  #endif
202 #endif
203  ak_log_set_message( error_event_string );
204  return ak_error_set_value( code );
205 }
