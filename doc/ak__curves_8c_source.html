41  void ak_mpzn_set_wcurve_discriminant( ak_uint64 *d, ak_wcurve ec )
42 {
43  ak_mpznmax s, one = ak_mpznmax_one;
44 
45  /* определяем константы 4 и 27 в представлении Монтгомери */
46  ak_mpzn_set_ui( d, ec->size, 4 );
47  ak_mpzn_set_ui( s, ak_mpznmax_size, 27 );
48  ak_mpzn_mul_montgomery( d, d, ec->r2, ec->p, ec->n, ec->size );
49  ak_mpzn_mul_montgomery( s, s, ec->r2, ec->p, ec->n, ec->size );
50 
51  /* вычисляем 4a^3 (mod p) значение в представлении Монтгомери */
52  ak_mpzn_mul_montgomery( d, d, ec->a, ec->p, ec->n, ec->size );
53  ak_mpzn_mul_montgomery( d, d, ec->a, ec->p, ec->n, ec->size );
54  ak_mpzn_mul_montgomery( d, d, ec->a, ec->p, ec->n, ec->size );
55 
56  /* вычисляем значение 4a^3 + 27b^2 (mod p) в представлении Монтгомери */
57  ak_mpzn_mul_montgomery( s, s, ec->b, ec->p, ec->n, ec->size );
58  ak_mpzn_mul_montgomery( s, s, ec->b, ec->p, ec->n, ec->size );
59  ak_mpzn_add_montgomery( d, d, s, ec->p, ec->size );
60 
61  /* определяем константу -16 в представлении Монтгомери и вычисляем D = -16(4a^3+27b^2) (mod p) */
62  ak_mpzn_set_ui( s, ec->size, 16 );
63  ak_mpzn_sub( s, ec->p, s, ec->size );
64  ak_mpzn_mul_montgomery( s, s, ec->r2, ec->p, ec->n, ec->size );
65  ak_mpzn_mul_montgomery( d, d, s, ec->p, ec->n, ec->size );
66 
67  /* возвращаем результат (в обычном представлении) */
68  ak_mpzn_mul_montgomery( d, d, one, ec->p, ec->n, ec->size );
69 }

98  int ak_wcurve_check_order_parameters( ak_wcurve ec )
99 {
100  ak_mpzn512 r, s, t;
101  struct random generator;
102 
103  ak_random_create_lcg( &generator );
104  ak_mpzn_set_random( t, ec->size, &generator );
105  ak_mpzn_rem( t, t, ec->q, ec->size );
106  ak_random_destroy( &generator );
107 
108  ak_mpzn_set_ui( r, ec->size, 2 );
109  ak_mpzn_sub( r, ec->q, r, ec->size );
110  ak_mpzn_modpow_montgomery( s, t, r, ec->q, ec->nq, ec->size );
111  ak_mpzn_mul_montgomery( t, s, t, ec->q, ec->nq, ec->size );
112 
113  ak_mpzn_mul_montgomery( t, t, ec->r2q, ec->q, ec->nq, ec->size );
114  ak_mpzn_mul_montgomery( t, t, ec->point.z, ec->q, ec->nq, ec->size );
115  ak_mpzn_mul_montgomery( t, t, ec->point.z, ec->q, ec->nq, ec->size );
116  if( ak_mpzn_cmp_ui( t, ec->size, 1 )) return ak_error_ok;
117  else return ak_error_curve_order_parameters;
118 }

140  int ak_wcurve_is_ok( ak_wcurve ec )
141 {
142  int error = ak_error_ok;
143  struct wpoint wp;
144 
145  /* создали кривую и проверяем ее параметры */
146  if( ec->p[ ec->size-1 ] < 0x100000000LL )
147  return ak_error_message( ak_error_curve_prime_size, __func__ ,
148  "using elliptic curve parameters with wrong module" );
149  if(( error = ak_wcurve_discriminant_is_ok( ec )) != ak_error_ok )
150  return ak_error_message( ak_error_curve_discriminant, __func__ ,
151  "using elliptic curve parameters with zero discriminant" );
152  /* теперь тестируем точку на кривой */
153  if(( error = ak_wpoint_set( &wp, ec )) != ak_error_ok )
154  return ak_error_message( error, __func__, "incorect asiigning a temporary point" );
155  if( ak_wpoint_is_ok( &wp, ec ) != ak_true )
156  return ak_error_message( ak_error_curve_point, __func__ ,
157  "elliptic curve parameters has'nt correct point" );
158  if( ak_wpoint_check_order( &wp, ec ) != ak_true )
159  return ak_error_message( ak_error_curve_point_order, __func__ ,
160  "elliptic curve point has wrong order" );
161  /* тестируем параметры порядка группы точек, используемые для выработки и проверки электронной подписи */
162  if(( error = ak_wcurve_check_order_parameters( ec )) != ak_error_ok )
163  return ak_error_message( error, __func__ ,
164  "elliptic curve has wrong parameters for calculation in prime field modulo q" );
165  return error;
166 }

219  ak_bool ak_wcurve_test( void )
220 {
221  ak_oid oid = NULL;
222  ak_bool result = ak_true;
223  int reason = ak_error_ok, audit = ak_log_get_level();
224 
225  if( audit >= ak_log_maximum )
226  ak_error_message( ak_error_ok, __func__ , "testing Weierstrass curves started" );
227 
228  /* организуем цикл по перебору всех известных библиотеке параметров эллиптических кривых */
229  oid = ak_oid_find_by_engine( identifier );
230  while( oid != NULL ) {
231  if( oid->mode == wcurve_params ) {
232  ak_wcurve wc = NULL;
233  if(( wc = ( ak_wcurve ) oid->data ) == NULL ) {
234  ak_error_message( ak_error_null_pointer, __func__,
235  "internal error with null poionter to wcurve paramset" );
236  result = ak_false;
237  goto lab_exit;
238  }
239  if(( reason = ak_wcurve_is_ok( wc )) != ak_error_ok ) {
240  char *p = NULL;
241  switch( reason ) {
242  case ak_error_curve_discriminant : p = "discriminant"; break;
243  case ak_error_curve_point : p = "base point"; break;
244  case ak_error_curve_point_order : p = "base point order"; break;
245  case ak_error_curve_prime_size : p = "prime modulo p"; break;
246  case ak_error_curve_order_parameters : p = "prime order parameters"; break;
247  default : p = "unexpected parameter";
248  }
249  ak_wcurve_to_log( wc );
250  ak_error_message_fmt( reason, __func__ , "curve %s (OID: %s) has wrong %s",
251  oid->name, oid->id, p );
252  result = ak_false;
253  goto lab_exit;
254  } else
255  if( audit > ak_log_standard ) {
256  ak_error_message_fmt( ak_error_ok, __func__ , "curve %s (OID: %s) is Ok",
257  oid->name, oid->id );
258  }
259  }
260  oid = ak_oid_findnext_by_engine( oid, identifier );
261  }
262 
263  lab_exit:
264  if( !result ) ak_error_message( ak_error_get_value(), __func__ ,
265  "incorrect testing Weierstrass curves" );
266  else if( audit >= ak_log_maximum ) ak_error_message( ak_error_get_value(), __func__ ,
267  "testing Weierstrass curves ended successfully" );
268  return result;
269 }

345  ak_bool ak_wpoint_is_ok( ak_wpoint wp, ak_wcurve ec )
346 {
347  ak_mpznmax t, s;
348  memset( t, 0, sizeof(ak_uint64)*ak_mpznmax_size );
349  memset( s, 0, sizeof(ak_uint64)*ak_mpznmax_size );
350 
351  /* Проверяем принадлежность точки заданной кривой */
352  ak_mpzn_set( t, ec->a, ec->size );
353  ak_mpzn_mul_montgomery( t, t, wp->x, ec->p, ec->n, ec->size );
354  ak_mpzn_set( s, ec->b, ec->size );
355  ak_mpzn_mul_montgomery( s, s, wp->z, ec->p, ec->n, ec->size );
356  ak_mpzn_add_montgomery( t, t, s, ec->p, ec->size ); // теперь в t величина (ax+bz)
357 
358  ak_mpzn_set( s, wp->z, ec->size );
359  ak_mpzn_mul_montgomery( s, s, s, ec->p, ec->n, ec->size );
360  ak_mpzn_mul_montgomery( t, t, s, ec->p, ec->n, ec->size ); // теперь в t величина (ax+bz)z^2
361 
362  ak_mpzn_set( s, wp->x, ec->size );
363  ak_mpzn_mul_montgomery( s, s, s, ec->p, ec->n, ec->size );
364  ak_mpzn_mul_montgomery( s, s, wp->x, ec->p, ec->n, ec->size );
365  ak_mpzn_add_montgomery( t, t, s, ec->p, ec->size ); // теперь в t величина x^3 + (ax+bz)z^2
366 
367  ak_mpzn_set( s, wp->y, ec->size );
368  ak_mpzn_mul_montgomery( s, s, s, ec->p, ec->n, ec->size );
369  ak_mpzn_mul_montgomery( s, s, wp->z, ec->p, ec->n, ec->size ); // теперь в s величина x^3 + (ax+bz)z^2
370 
371  if( ak_mpzn_cmp( t, s, ec->size )) return ak_false;
372  return ak_true;
373 }

401  void ak_wpoint_double( ak_wpoint wp, ak_wcurve ec )
402 {
403  ak_mpznmax u1, u2, u3, u4, u5, u6, u7;
404 
405  if( ak_mpzn_cmp_ui( wp->z, ec->size, 0 ) == ak_true ) return;
406  if( ak_mpzn_cmp_ui( wp->y, ec->size, 0 ) == ak_true ) {
407  ak_wpoint_set_as_unit( wp, ec );
408  return;
409  }
410  // dbl-2007-bl
411  ak_mpzn_mul_montgomery( u1, wp->x, wp->x, ec->p, ec->n, ec->size );
412  ak_mpzn_mul_montgomery( u2, wp->z, wp->z, ec->p, ec->n, ec->size );
413  ak_mpzn_lshift_montgomery( u4, u1, ec->p, ec->size );
414  ak_mpzn_add_montgomery( u4, u4, u1, ec->p, ec->size );
415  ak_mpzn_mul_montgomery( u3, u2, ec->a, ec->p, ec->n, ec->size );
416  ak_mpzn_add_montgomery( u3, u3, u4, ec->p, ec->size ); // u3 = az^2 + 3x^2
417  ak_mpzn_mul_montgomery( u4, wp->y, wp->z, ec->p, ec->n, ec->size );
418  ak_mpzn_lshift_montgomery( u4, u4, ec->p, ec->size ); // u4 = 2yz
419  ak_mpzn_mul_montgomery( u5, wp->y, u4, ec->p, ec->n, ec->size ); // u5 = 2y^2z
420  ak_mpzn_lshift_montgomery( u6, u5, ec->p, ec->size ); // u6 = 2u5
421  ak_mpzn_mul_montgomery( u7, u6, wp->x, ec->p, ec->n, ec->size ); // u7 = 8xy^2z
422  ak_mpzn_lshift_montgomery( u1, u7, ec->p, ec->size );
423  ak_mpzn_sub( u1, ec->p, u1, ec->size );
424  ak_mpzn_mul_montgomery( u2, u3, u3, ec->p, ec->n, ec->size );
425  ak_mpzn_add_montgomery( u2, u2, u1, ec->p, ec->size );
426  ak_mpzn_mul_montgomery( wp->x, u2, u4, ec->p, ec->n, ec->size );
427  ak_mpzn_mul_montgomery( u6, u6, u5, ec->p, ec->n, ec->size );
428  ak_mpzn_sub( u6, ec->p, u6, ec->size );
429  ak_mpzn_sub( u2, ec->p, u2, ec->size );
430  ak_mpzn_add_montgomery( u2, u2, u7, ec->p, ec->size );
431  ak_mpzn_mul_montgomery( wp->y, u2, u3, ec->p, ec->n, ec->size );
432  ak_mpzn_add_montgomery( wp->y, wp->y, u6, ec->p, ec->size );
433  ak_mpzn_mul_montgomery( wp->z, u4, u4, ec->p, ec->n, ec->size );
434  ak_mpzn_mul_montgomery( wp->z, wp->z, u4, ec->p, ec->n, ec->size );
435 }

471  void ak_wpoint_add( ak_wpoint wp1, ak_wpoint wp2, ak_wcurve ec )
472 {
473  ak_mpznmax u1, u2, u3, u4, u5, u6, u7;
474 
475  if( ak_mpzn_cmp_ui( wp2->z, ec->size, 0 ) == ak_true ) return;
476  if( ak_mpzn_cmp_ui( wp1->z, ec->size, 0 ) == ak_true ) {
477  ak_wpoint_set_wpoint( wp1, wp2, ec );
478  return;
479  }
480  // поскольку удвоение точки с помощью формул сложения дает бесконечно удаленную точку,
481  // необходимо выполнить проверку
482  ak_mpzn_mul_montgomery( u1, wp1->x, wp2->z, ec->p, ec->n, ec->size );
483  ak_mpzn_mul_montgomery( u2, wp2->x, wp1->z, ec->p, ec->n, ec->size );
484  if( ak_mpzn_cmp( u1, u2, ec->size ) == 0 ) { // случай совпадения х-координат точки
485  ak_mpzn_mul_montgomery( u1, wp1->y, wp2->z, ec->p, ec->n, ec->size );
486  ak_mpzn_mul_montgomery( u2, wp2->y, wp1->z, ec->p, ec->n, ec->size );
487  if( ak_mpzn_cmp( u1, u2, ec->size ) == 0 ) // случай полного совпадения точек
488  ak_wpoint_double( wp1, ec );
489  else ak_wpoint_set_as_unit( wp1, ec );
490  return;

567  void ak_wpoint_pow( ak_wpoint wq, ak_wpoint wp, ak_uint64 *k, size_t size, ak_wcurve ec )
568 {
569  ak_uint64 uk = 0;
570  long long int i, j;
571  struct wpoint Q, R; /* две точки из лесенки Монтгомери */
572 
573  /* начальные значения для переменных */
574  ak_wpoint_set_as_unit( &Q, ec );
575  ak_wpoint_set_wpoint( &R, wp, ec );
576 
577  /* полный цикл по всем(!) битам числа k */
578  for( i = size-1; i >= 0; i-- ) {
579  uk = k[i];
580  for( j = 0; j < 64; j++ ) {
581  if( uk&0x8000000000000000LL ) { ak_wpoint_add( &Q, &R, ec ); ak_wpoint_double( &R, ec ); }
582  else { ak_wpoint_add( &R, &Q, ec ); ak_wpoint_double( &Q, ec ); }
583  uk <<= 1;
584  }
585  }
586  /* копируем полученный результат */
587  ak_wpoint_set_wpoint( wq, &Q, ec );
588 }
