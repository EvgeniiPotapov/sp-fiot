41  int ak_hash_create( ak_hash ctx, const size_t data_size, const size_t block_size )
42 {
43  if( ctx == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
44  "using null pointer to hash context" );
45  if( block_size == 0 ) return ak_error_message( ak_error_zero_length, __func__ ,
46  "using a zero length of data block length" );
47  if( data_size != 0 ) {
48  if(( ctx->data = malloc( data_size )) == NULL )
49  return ak_error_message( ak_error_out_of_memory, __func__ ,
50  "incorrect internal data memory allocation" );
51  } else ctx->data = NULL;
52 
53  ctx->bsize = block_size;
54  ctx->hsize = 0;
55  ctx->oid = NULL;
56  ctx->clean = NULL;
57  ctx->update = NULL;
58  ctx->finalize = NULL;
59 
60  return ak_error_ok;
61 }

113  int ak_hash_create_oid( ak_hash ctx, ak_oid oid )
114 {
115  int error = ak_error_ok;
116 
117  /* выполняем проверку */
118  if( ctx == NULL ) return ak_error_message( ak_error_null_pointer, __func__,
119  "using null pointer to hash context" );
120  if( oid == NULL ) return ak_error_message( ak_error_null_pointer, __func__,
121  "using null pointer to hash function OID" );
122  /* проверяем, что OID от бесключевой функции хеширования */
123  if( oid->engine != hash_function )
124  return ak_error_message( ak_error_oid_engine, __func__ , "using oid with wrong engine" );
125  /* проверяем, что OID от алгоритма, а не от параметров */
126  if( oid->mode != algorithm )
127  return ak_error_message( ak_error_oid_mode, __func__ , "using oid with wrong mode" );
128  /* проверяем, что производящая функция определена */
129  if( oid->func == NULL )
130  return ak_error_message( ak_error_undefined_function, __func__ ,
131  "using oid with undefined constructor function" );
132  /* инициализируем контекст */
133  if(( error = (( ak_function_hash_create *)oid->func)( ctx )) != ak_error_ok )
134  return ak_error_message( error, __func__, "invalid creation of hash function context");
135 
136  return error;
137 }

158  ak_buffer ak_hash_context_ptr( ak_hash ctx, const ak_pointer in, const size_t size, ak_pointer out )
159 {
160  ak_buffer result = NULL;
161  size_t quot = 0, offset = 0;
162 
163  if( ctx == NULL ) {
164  ak_error_message( ak_error_null_pointer, __func__ , "using null pointer to hash context" );
165  return NULL;
166  }
167  if( in == NULL ) {
168  ak_error_message( ak_error_null_pointer, __func__ , "using null pointer to data" );
169  return NULL;
170  }
171 
172  /* вычищаем результаты предыдущих вычислений */
173  ctx->clean( ctx );
174  quot = size/ctx->bsize;
175  offset = quot*ctx->bsize;
176  /* вызываем, если длина сообщения не менее одного полного блока */
177  if( quot > 0 ) ctx->update( ctx, in, offset );
178  /* обрабатываем хвост */
179  result = ctx->finalize( ctx, (unsigned char *)in + offset, size - offset, out );
180  /* очищаем за собой данные, содержащиеся в контексте */
181  ctx->clean( ctx );
182  return result;
183 }

237  ak_handle ak_hash_new_gosthash94( ak_handle handle )
238 {
239  ak_hash ctx = NULL;
240  int error = ak_error_ok;
241  ak_oid oid = ak_handle_get_context( handle, oid_engine );
242 
243  /* получаем oid таблиц замен (указатель на данные) */
244  if( oid == NULL ) {
245  ak_error_message( ak_error_get_value(), __func__ , "using broken OID" );
246  return ak_error_wrong_handle;
247  }
248  if( oid->engine != hash_function ) {
249  ak_error_message( ak_error_oid_engine, __func__ , "using not hash function OID" );
250  return ak_error_wrong_handle;
251  }
252  if( oid->mode != kbox_params ) {
253  ak_error_message( ak_error_oid_mode, __func__ , "using a wrong mode hash function OID" );
254  return ak_error_wrong_handle;
255  }
256 
257  /* создаем контекст функции хэширования */
258  if(( ctx = malloc( sizeof( struct hash ))) == NULL ) {
259  ak_error_message( ak_error_out_of_memory, __func__ , "wrong creation of hash function context" );
260  return ak_error_wrong_handle;
261  }
262 
263  /* инициализируем его */
264  if(( error = ak_hash_create_gosthash94( ctx, oid )) != ak_error_ok ) {
265  ak_error_message( error, __func__ , "wrong initialization of hash function context" );
266  free( ctx );
267  return ak_error_wrong_handle;
268  }
269 
270  /* помещаем в стуктуру управления контекстами */
271  return ak_libakrypt_new_handle( ctx, hash_function, "", ak_hash_delete );
272 }
