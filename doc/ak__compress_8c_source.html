55  int ak_compress_create_hash( ak_compress comp, ak_hash hctx )
56 {
57  /* вначале, необходимые проверки */
58  if( comp == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
59  "using null pointer to compress context" );
60  if( hctx == NULL ) return ak_error_message( ak_error_null_pointer, __func__ ,
61  "using null pointer to hash context" );
62  if(( hctx->update == NULL ) || ( hctx->finalize == NULL ) || ( hctx->clean == NULL ))
63  return ak_error_message( ak_error_undefined_function, __func__ ,
64  "using non initialized hash context" );
65  /* теперь собственно инициализация */
66  if(( comp->data = (ak_uint8 *) malloc( comp->bsize = hctx->bsize )) == NULL ) {
67  ak_error_message( ak_error_out_of_memory, __func__ ,
68  "wrong memory alllocation for a new temporary data buffer" );
69  } else memset( comp->data, 0, hctx->bsize );
70  comp->length = 0;
71 
72  /* устанавливаем значения и полей и методы из контекста функции хеширования */
73  comp->ctx = hctx;
74  comp->hsize = hctx->hsize;
75  comp->clean = hctx->clean;
76  comp->update = hctx->update;
77  comp->finalize = hctx->finalize;
78 
79  return ak_error_ok;
80 }

167  int ak_compress_update( ak_compress comp, const ak_pointer in, const size_t size )
168 {
169  ak_uint8 *ptrin = (ak_uint8 *) in;
170  size_t quot = 0, offset = 0, newsize = size;
171 
172  if( comp == NULL ) return ak_error_message( ak_error_null_pointer, __func__,
173  "using a null pointer to null compress context" );
174  if( comp->update == NULL ) return ak_error_message( ak_error_undefined_function, __func__ ,
175  "using an undefined update function" );
176 
177  /* в начале проверяем, есть ли данные во временном буфере */
178  if( comp->length != 0 ) {
179  /* если новых данных мало, то добавляем во временный буффер и выходим */
180  if(( comp->length + newsize ) < comp->bsize ) {
181  memcpy( comp->data + comp->length, ptrin, newsize );
182  comp->length += newsize;
183  return ak_error_ok;
184  }
185  /* дополняем буффер до длины, кратной bsize */
186  offset = comp->bsize - comp->length;
187  memcpy( comp->data + comp->length, ptrin, offset );
188 
189  /* обновляем значение контекста функции и очищаем временный буффер */
190  comp->update( comp->ctx, comp->data, comp->bsize );
191  memset( comp->data, 0, comp->bsize );
192  comp->length = 0;
193  ptrin += offset;
194  newsize -= offset;
195  }
196 
197  /* теперь обрабатываем входные данные с пустым временным буффером */
198  if( newsize != 0 ) {
199  quot = newsize/comp->bsize;
200  offset = quot*comp->bsize;
201  /* обрабатываем часть, кратную величине bsize */
202  if( quot > 0 ) comp->update( comp->ctx, ptrin, offset );
203  /* хвост оставляем на следующий раз */
204  if( offset < newsize ) {
205  comp->length = newsize - offset;
206  memcpy( comp->data, ptrin + offset, comp->length );
207  }
208  }
209 
210  return ak_error_ok;
211 }

234  ak_buffer ak_compress_finalize( ak_compress comp,
235  const ak_pointer in, const size_t size, ak_pointer out )
236 {
237  if( comp == NULL ) {
238  ak_error_message( ak_error_null_pointer, __func__,
239  "using a null pointer to null compress context" );
240  return NULL;
241  }
242  if( comp->finalize == NULL ) {
243  ak_error_message( ak_error_undefined_function, __func__ ,
244  "using an undefined finalize function" );
245  return NULL;
246  }
247 
248  /* начинаем с того, что обрабатываем все переданные данные */
249  ak_compress_update( comp, in, size );
250  /* потом обрабатываем хвост, оставшийся во временном буффере, и выходим */
251  return comp->finalize( comp->ctx, comp->data, comp->length, out );
252 }

271  ak_buffer ak_compress_file( ak_compress comp, const char* filename, ak_pointer out )
272 {
273  int fd = 0;
274  struct stat st;
275  size_t len = 0;
276  ak_uint8 *localbuffer; /* место для локального считывания информации */
277  size_t block_size = 4096; /* оптимальная длина блока для Windows пока не ясна */
278  ak_buffer result = NULL;
279 
280  /* выполняем необходимые проверки */
281  if( comp == NULL ) {
282  ak_error_message( ak_error_null_pointer, __func__ , "use a null pointer to compress context" );
283  return NULL;
284  }
285  if( filename == NULL ) {
286  ak_error_message( ak_error_null_pointer, __func__ , "use a null pointer to filename" );
287  return NULL;
288  }
289 
290  if(( fd = open( filename, O_RDONLY | O_BINARY )) < 0 ) {
291  ak_error_message( ak_error_open_file, __func__, strerror( errno ));
292  return NULL;
293  }
294  if( fstat( fd, &st ) ) {
295  close( fd );
296  ak_error_message( ak_error_access_file, __func__, strerror( errno ));
297  return NULL;
298  }
299 
300  /* для файла нулевой длины результатом будет хеш от нулевого вектора */
301  ak_compress_clean( comp );
302  if( !st.st_size ) return ak_compress_finalize( comp, "", 0, out );
303 
304  /* готовим область для хранения данных */
305  #ifdef _WIN32
306  block_size = ak_max( 4096, comp->bsize );
307  #else
308  block_size = ak_max( st.st_blksize, comp->bsize );
309  #endif
310  /* здесь мы выделяем локальный буффер для считывания/обработки данных */
311  if((localbuffer = ( ak_uint8 * ) malloc( block_size )) == NULL ) {
312  close( fd );
313  ak_error_message( ak_error_out_of_memory, __func__ , "out of memory" );
314  return NULL;
315  }
316  /* теперь обрабатываем файл с данными */
317  #ifdef _WIN32
318  read_label: len = read( fd, localbuffer, (unsigned int) block_size );
319  #else
320  read_label: len = read( fd, localbuffer, block_size );
321  #endif
322  if( len == block_size ) {
323  ak_compress_update( comp, localbuffer, block_size ); /* добавляем считанные данные */
324  goto read_label;
325  } else {
326  size_t qcnt = len / comp->bsize,
327  tail = len - qcnt*comp->bsize;
328  if( qcnt ) ak_compress_update( comp, localbuffer, qcnt*comp->bsize );
329  result = ak_compress_finalize( comp, localbuffer + qcnt*comp->bsize, tail, out );
330  }
331  /* очищаем за собой данные, содержащиеся в контексте */
332  ak_compress_clean( comp );
333  /* закрываем данные */
334  close(fd);
335  free( localbuffer );
336  return result;
337 }
